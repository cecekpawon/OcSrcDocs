214PatcherGetSymbolAddress (
215 IN OUT PATCHER_CONTEXT *Context,
216 IN CONST CHAR8 *Name,
217 IN OUT UINT8 **Address
218 )
219{
220 MACH_NLIST_ANY *Symbol;
221 CONST CHAR8 *SymbolName;
222 UINT64 SymbolAddress;
223 UINT32 Offset;
224 UINT32 Index;
225
226 Index = 0;
227 Offset = 0;
228 while (TRUE) {
229 //
230 // Try the usual way first via SYMTAB.
231 //
232 Symbol = MachoGetSymbolByIndex (&Context->MachContext, Index);
233 if (Symbol == NULL) {
234 //
235 // If we have KxldState, use it.
236 //
237 if ((Index == 0) && (Context->KxldState != NULL)) {
238 SymbolAddress = InternalKxldSolveSymbol (
239 Context->Is32Bit,
240 Context->KxldState,
241 Context->KxldStateSize,
242 Name
243 );
244 //
245 // If we have a symbol, get its ondisk offset.
246 //
247 if ((SymbolAddress != 0) && MachoSymbolGetDirectFileOffset (&Context->MachContext, SymbolAddress, &Offset, NULL)) {
248 //
249 // Proceed to success.
250 //
251 break;
252 }
253 }
254
255 return EFI_NOT_FOUND;
256 }
257
258 SymbolName = MachoGetSymbolName (&Context->MachContext, Symbol);
259 if ((SymbolName != NULL) && (AsciiStrCmp (Name, SymbolName) == 0)) {
260 //
261 // Once we have a symbol, get its ondisk offset.
262 //
263 if (MachoSymbolGetFileOffset (&Context->MachContext, Symbol, &Offset, NULL)) {
264 //
265 // Proceed to success.
266 //
267 break;
268 }
269
270 return EFI_INVALID_PARAMETER;
271 }
272
273 Index++;
274 }

276 *Address = (UINT8 *)MachoGetMachHeader (&Context->MachContext) + Offset;
277 return EFI_SUCCESS;

502EFI_STATUS
503PatcherBlockKext (
504 IN OUT PATCHER_CONTEXT *Context
505 )
506{
507 UINT8 *MachBase;
508 UINT32 MachSize;
509 UINT64 KmodOffset;
510 UINT64 StartAddr;
511 UINT64 TmpOffset;
512 UINT8 *KmodInfo;
513 UINT8 *PatchAddr;
514
515 //
516 // Kernel has 0 kmod.
517 //
518 if ((Context->VirtualKmod == 0) || (Context->VirtualBase > Context->VirtualKmod)) {
519 return EFI_UNSUPPORTED;
520 }
521
522 MachBase = (UINT8 *)MachoGetMachHeader (&Context->MachContext);
523 MachSize = MachoGetFileSize (&Context->MachContext);
524
525 //
526 // Determine offset of kmod within file.
527 //
528 KmodOffset = Context->VirtualKmod - Context->VirtualBase;
529 if ( OcOverflowAddU64 (KmodOffset, Context->FileOffset, &KmodOffset)
530 || OcOverflowAddU64 (KmodOffset, Context->Is32Bit ? sizeof (KMOD_INFO_32_V1) : sizeof (KMOD_INFO_64_V1), &TmpOffset)
531 || (TmpOffset > MachSize))
532 {
533 return EFI_INVALID_PARAMETER;
534 }
535
536 KmodInfo = MachBase + KmodOffset;
537 if (Context->Is32Bit) {
538 StartAddr = ((KMOD_INFO_32_V1 *)KmodInfo)->StartAddr;
539 } else {
540 StartAddr = ((KMOD_INFO_64_V1 *)KmodInfo)->StartAddr;
541 if (Context->IsKernelCollection) {
542 StartAddr = KcFixupValue (StartAddr, NULL);
543 }
544 }
545
546 if ((StartAddr == 0) || (Context->VirtualBase > StartAddr)) {
547 return EFI_INVALID_PARAMETER;
548 }
549
550 TmpOffset = StartAddr - Context->VirtualBase;
551 if ( OcOverflowAddU64 (TmpOffset, Context->FileOffset, &TmpOffset)
552 || (TmpOffset > MachSize - 6))
553 {
554 return EFI_BUFFER_TOO_SMALL;
555 }
556
557 DEBUG ((
558 DEBUG_VERBOSE,
559 "OCAK: %a-bit blocker start @ 0x%llX\n",
560 Context->Is32Bit ? "32" : "64",
561 TmpOffset
562 ));
563
564 PatchAddr = MachBase + TmpOffset;
565
566 //
567 // mov eax, KMOD_RETURN_FAILURE
568 // ret
569 //
570 PatchAddr[0] = 0xB8;
571 PatchAddr[1] = KMOD_RETURN_FAILURE;
572 PatchAddr[2] = 0x00;
573 PatchAddr[3] = 0x00;
574 PatchAddr[4] = 0x00;
575 PatchAddr[5] = 0xC3;
576
577 return EFI_SUCCESS;
578}
