31// Macro for building verbs. 4-bit verbs will be zero when masked against 0xFF0.
32#define HDA_CODEC_VERB(Verb, Payload) ((UINT32)((((Verb) & 0xFFF) & 0xFF0) ? \
33 ((((Verb) & 0xFFF) << 8) | ((Payload) & 0xFF)) : /* 12-bit verbs */ \
34 ((((Verb) & 0xF) << 16) | ((Payload) & 0xFFFF)))) /* 4-bit verbs */

47#define HDA_CONVERTER_FORMAT_MULT(a) ((UINT8)(((a) >> 11) & 0x3))
48#define HDA_CONVERTER_FORMAT_BASE_44KHZ BIT14
49#define HDA_CONVERTER_FORMAT_SET(chan, bits, div, mult, base) \
50 ((UINT16)(((chan) & 0xF) | (((bits) & 0x3) << 4) | (((div) & 0x3) << 8) | \
