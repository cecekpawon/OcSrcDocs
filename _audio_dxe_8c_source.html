65EFI_STATUS
66EFIAPI
67AudioDxeInit (
68 IN EFI_HANDLE ImageHandle,
69 IN EFI_SYSTEM_TABLE *SystemTable
70 )
71{
72 EFI_STATUS Status;
73 EFI_LOADED_IMAGE_PROTOCOL *LoadedImage;
74 OC_FLEX_ARRAY *ParsedLoadOptions;
75
76 //
77 // Parse optional driver params.
78 //
79 Status = gBS->HandleProtocol (
80 ImageHandle,
81 &gEfiLoadedImageProtocolGuid,
82 (VOID **)&LoadedImage
83 );
84
85 if (EFI_ERROR (Status)) {
86 return Status;
87 }
88
89 Status = OcParseLoadOptions (LoadedImage, &ParsedLoadOptions);
90 if (!EFI_ERROR (Status)) {
91 gRestoreNoSnoop = OcHasParsedVar (ParsedLoadOptions, L"--restore-nosnoop", TRUE);
92
93 Status = OcParsedVarsGetInt (ParsedLoadOptions, L"--gpio-setup", &gGpioSetupStageMask, TRUE);
94 if ((Status == EFI_NOT_FOUND) && OcHasParsedVar (ParsedLoadOptions, L"--gpio-setup", TRUE)) {
95 gGpioSetupStageMask = GPIO_SETUP_STAGE_ALL;
96 }
97
98 DEBUG ((DEBUG_INFO, "HDA: GPIO setup stages 0x%X, restore NSNPEN %d\n", gGpioSetupStageMask, gRestoreNoSnoop));
99
100 if (gGpioSetupStageMask != GPIO_SETUP_STAGE_NONE) {
101 OcParsedVarsGetInt (ParsedLoadOptions, L"--gpio-pins", &gGpioPinMask, TRUE);
102 DEBUG ((
103 DEBUG_INFO,
104 "HDA: GPIO pin mask 0x%X%a\n",
105 gGpioPinMask,
106 gGpioPinMask == 0 ? " (auto)" : ""
107 ));
108 }
109
110 OcFlexArrayFree (&ParsedLoadOptions);
111 } else if (Status != EFI_NOT_FOUND) {
112 return Status;
113 }
114
115 //
116 // Register HdaController Driver Binding.
117 //
118 Status = EfiLibInstallDriverBindingComponentName2 (
119 ImageHandle,
120 SystemTable,
121 &gHdaControllerDriverBinding,
122 ImageHandle,
123 &gHdaControllerComponentName,
124 &gHdaControllerComponentName2
125 );
126
127 if (EFI_ERROR (Status)) {
128 return Status;
129 }
130
131 //
132 // Register HdaCodec Driver Binding.
133 //
134 Status = EfiLibInstallDriverBindingComponentName2 (
135 ImageHandle,
136 SystemTable,
137 &gHdaCodecDriverBinding,
138 NULL,
139 &gHdaCodecComponentName,
140 &gHdaCodecComponentName2
141 );
142
143 ASSERT_EFI_ERROR (Status);
144 if (EFI_ERROR (Status)) {
145 return Status;
146 }
147
148 Status = gBS->InstallMultipleProtocolInterfaces (
149 &ImageHandle,
150 &gEfiAudioDecodeProtocolGuid,
151 &gEfiAudioDecodeProtocol,
152 NULL
153 );
154
155 ASSERT_EFI_ERROR (Status);
156 if (EFI_ERROR (Status)) {
157 return Status;
158 }
159
160 return EFI_SUCCESS;
161}
