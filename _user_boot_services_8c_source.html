84 //
85 // NOTE: This code is adapted from original EDK II implementations.
86 //
87
88 UINTN Index;
89 EFI_CONFIGURATION_TABLE *EfiConfigurationTable;
90 EFI_CONFIGURATION_TABLE *OldTable;
91
92 //
93 // If Guid is NULL, then this operation cannot be performed
94 //
95 if (Guid == NULL) {
96 return EFI_INVALID_PARAMETER;
97 }
98
99 EfiConfigurationTable = gST->ConfigurationTable;
100
101 //
102 // Search all the table for an entry that matches Guid
103 //
104 for (Index = 0; Index < gST->NumberOfTableEntries; Index++) {
105 if (CompareGuid (Guid, &(gST->ConfigurationTable[Index].VendorGuid))) {
106 break;
107 }
108 }
109
110 if (Index < gST->NumberOfTableEntries) {
111 //
112 // A match was found, so this is either a modify or a delete operation
113 //
114 if (Table != NULL) {
115 //
116 // If Table is not NULL, then this is a modify operation.
117 // Modify the table entry and return.
118 //
119 gST->ConfigurationTable[Index].VendorTable = Table;
120
121 return EFI_SUCCESS;
122 }
123
124 //
125 // A match was found and Table is NULL, so this is a delete operation.
126 //
127 gST->NumberOfTableEntries--;
128
129 //
130 // Copy over deleted entry
131 //
132 CopyMem (
133 &(EfiConfigurationTable[Index]),
134 &(gST->ConfigurationTable[Index + 1]),
135 (gST->NumberOfTableEntries - Index) * sizeof (EFI_CONFIGURATION_TABLE)
136 );
137 } else {
138 //
139 // No matching GUIDs were found, so this is an add operation.
140 //
141 if (Table == NULL) {
142 //
143 // If Table is NULL on an add operation, then return an error.
144 //
145 return EFI_NOT_FOUND;
146 }
147
148 //
149 // Assume that Index == gST->NumberOfTableEntries
150 //
151 if ((Index * sizeof (EFI_CONFIGURATION_TABLE)) >= mSystemTableAllocateSize) {
152 //
153 // Allocate a table with one additional entry.
154 //
155 mSystemTableAllocateSize += (CONFIG_TABLE_SIZE_INCREASED * sizeof (EFI_CONFIGURATION_TABLE));
156 EfiConfigurationTable = AllocatePool (mSystemTableAllocateSize);
157 if (EfiConfigurationTable == NULL) {
158 //
159 // If a new table could not be allocated, then return an error.
160 //
161 return EFI_OUT_OF_RESOURCES;
162 }
163
164 if (gST->ConfigurationTable != NULL) {
165 //
166 // Copy the old table to the new table.
167 //
168 CopyMem (
169 EfiConfigurationTable,
170 gST->ConfigurationTable,
171 Index * sizeof (EFI_CONFIGURATION_TABLE)
172 );
173
174 //
175 // Record the old table pointer.
176 //
177 OldTable = gST->ConfigurationTable;
178
179 //
180 // As the CoreInstallConfigurationTable() may be re-entered by CoreFreePool()
181 // in its calling stack, updating System table to the new table pointer must
182 // be done before calling CoreFreePool() to free the old table.
183 // It can make sure the gST->ConfigurationTable point to the new table
184 // and avoid the errors of use-after-free to the old table by the reenter of
185 // CoreInstallConfigurationTable() in CoreFreePool()'s calling stack.
186 //
187 gST->ConfigurationTable = EfiConfigurationTable;
188
189 //
190 // Free the old table after updating System Table to the new table pointer.
191 //
192 FreePool (OldTable);
193 } else {
194 //
195 // Update System Table
196 //
197 gST->ConfigurationTable = EfiConfigurationTable;
198 }
199 }
200
201 //
202 // Fill in the new entry
203 //
204 CopyGuid ((VOID *)&EfiConfigurationTable[Index].VendorGuid, Guid);
205 EfiConfigurationTable[Index].VendorTable = Table;
206
207 //
208 // This is an add operation, so increment the number of table entries
209 //
210 gST->NumberOfTableEntries++;
211 }
212
213 return EFI_SUCCESS;
214}
