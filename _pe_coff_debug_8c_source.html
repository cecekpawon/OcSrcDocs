229 CodeViewEntry = (EFI_IMAGE_DEBUG_DIRECTORY_ENTRY *)(VOID *)(
230 (CHAR8 *)Context->ImageBuffer + Context->CodeViewRva
231 );
232 //
233 // Load the Codeview information if present
234 //
235 if (CodeViewEntry->RVA != 0) {
236 Result = BaseOverflowAddU32 (
237 CodeViewEntry->RVA,
238 CodeViewEntry->SizeOfData,
239 &DebugEntryRvaTop
240 );
241 if ( Result || (DebugEntryRvaTop > Context->SizeOfImage)
242 || !IS_ALIGNED (CodeViewEntry->RVA, OC_ALIGNOF (UINT32)))
243 {
244 Context->CodeViewRva = 0;
245 return;
246 }
247 } else {
248 if ( !PcdGetBool (PcdImageLoaderForceLoadDebug)
249 || (Context->SizeOfImageDebugAdd == 0))
250 {
251 Context->CodeViewRva = 0;
252 return;
253 }
254
255 CodeViewEntry->RVA = ALIGN_VALUE (Context->SizeOfImage, OC_ALIGNOF (UINT32));
256
257 ASSERT (Context->SizeOfImageDebugAdd >= (CodeViewEntry->RVA - Context->SizeOfImage) + CodeViewEntry->SizeOfData);
258
259 CopyMem (
260 (CHAR8 *)Context->ImageBuffer + CodeViewEntry->RVA,
261 (CONST CHAR8 *)Context->FileBuffer + (CodeViewEntry->FileOffset - Context->TeStrippedOffset),
262 CodeViewEntry->SizeOfData
263 );
264 }
265
266 if (CodeViewEntry->SizeOfData < sizeof (UINT32)) {
267 Context->CodeViewRva = 0;
