110OcProcessVariableGuid (
111 IN CONST CHAR8 *AsciiVariableGuid,
112 OUT GUID *VariableGuid,
113 IN OC_NVRAM_LEGACY_MAP *Schema OPTIONAL,
114 OUT OC_NVRAM_LEGACY_ENTRY **SchemaEntry OPTIONAL
115 )
116{
117 EFI_STATUS Status;
118 UINT32 GuidIndex;
119
120 Status = AsciiStrToGuid (AsciiVariableGuid, VariableGuid);
121 if (EFI_ERROR (Status)) {
122 DEBUG ((DEBUG_WARN, "OC: Failed to convert NVRAM GUID %a - %r\n", AsciiVariableGuid, Status));
123 }
124
125 if (!EFI_ERROR (Status) && (Schema != NULL)) {
126 for (GuidIndex = 0; GuidIndex < Schema->Count; ++GuidIndex) {
127 if (AsciiStrCmp (AsciiVariableGuid, OC_BLOB_GET (Schema->Keys[GuidIndex])) == 0) {
128 *SchemaEntry = Schema->Values[GuidIndex];
129 return Status;
130 }
131 }
132
133 DEBUG ((DEBUG_INFO, "OC: Ignoring NVRAM GUID %a\n", AsciiVariableGuid));
134 Status = EFI_SECURITY_VIOLATION;
135 }
136
137 return Status;

141VOID
142OcSetNvramVariable (
143 IN CONST CHAR8 *AsciiVariableName,
144 IN EFI_GUID *VariableGuid,
145 IN UINT32 Attributes,
146 IN UINT32 VariableSize,
147 IN VOID *VariableData,
148 IN OC_NVRAM_LEGACY_ENTRY *SchemaEntry,
149 IN BOOLEAN Overwrite
150 )
151{
152 EFI_STATUS Status;
153 UINTN OriginalVariableSize;
154 CHAR16 *UnicodeVariableName;
155 BOOLEAN IsAllowed;
156 UINT32 VariableIndex;
157 VOID *OrgValue;
158 UINTN OrgSize;
159 UINT32 OrgAttributes;
160
161 if (SchemaEntry != NULL) {
162 IsAllowed = FALSE;
163
164 //
165 // TODO: Consider optimising lookup if it causes problems...
166 //
167 for (VariableIndex = 0; VariableIndex < SchemaEntry->Count; ++VariableIndex) {
168 if ((VariableIndex == 0) && (AsciiStrCmp ("*", OC_BLOB_GET (SchemaEntry->Values[VariableIndex])) == 0)) {
169 IsAllowed = TRUE;
170 break;
171 }
172
173 if (AsciiStrCmp (AsciiVariableName, OC_BLOB_GET (SchemaEntry->Values[VariableIndex])) == 0) {
174 IsAllowed = TRUE;
175 break;
176 }
177 }
178
179 if (!IsAllowed) {
180 DEBUG ((DEBUG_INFO, "OC: Setting NVRAM %g:%a is not permitted\n", VariableGuid, AsciiVariableName));
181 return;
182 }
183 }
184
185 UnicodeVariableName = AsciiStrCopyToUnicode (AsciiVariableName, 0);
186
187 if (UnicodeVariableName == NULL) {
188 DEBUG ((DEBUG_WARN, "OC: Failed to convert NVRAM variable name %a\n", AsciiVariableName));
189 return;
190 }
191
192 OriginalVariableSize = 0;
193 Status = gRT->GetVariable (
194 UnicodeVariableName,
195 VariableGuid,
196 NULL,
197 &OriginalVariableSize,
198 NULL
199 );
200
201 if ((Status == EFI_BUFFER_TOO_SMALL) && Overwrite) {
202 Status = GetVariable3 (UnicodeVariableName, VariableGuid, &OrgValue, &OrgSize, &OrgAttributes);
203 if (!EFI_ERROR (Status)) {
204 //
205 // Do not allow overwriting BS-only variables. Ideally we also check for NV attribute,
206 // but it is not set by Duet.
207 //
208 if ((Attributes & (EFI_VARIABLE_RUNTIME_ACCESS | EFI_VARIABLE_BOOTSERVICE_ACCESS))
209 == (EFI_VARIABLE_RUNTIME_ACCESS | EFI_VARIABLE_BOOTSERVICE_ACCESS))
210 {
211 Status = gRT->SetVariable (UnicodeVariableName, VariableGuid, 0, 0, 0);
212
213 if (EFI_ERROR (Status)) {
214 DEBUG ((
215 DEBUG_INFO,
216 "OC: Failed to delete overwritten variable %g:%a - %r\n",
217 VariableGuid,
218 AsciiVariableName,
219 Status
220 ));
221 Status = EFI_BUFFER_TOO_SMALL;
222 }
223 } else {
224 DEBUG ((
225 DEBUG_INFO,
226 "OC: Overwritten variable %g:%a has invalid attrs - %X\n",
227 VariableGuid,
228 AsciiVariableName,
229 Attributes
230 ));
231 Status = EFI_BUFFER_TOO_SMALL;
232 }
233
234 FreePool (OrgValue);
235 } else {
236 DEBUG ((
237 DEBUG_INFO,
238 "OC: Overwritten variable %g:%a has unknown attrs - %r\n",
239 VariableGuid,
240 AsciiVariableName,
241 Status
242 ));
243 Status = EFI_BUFFER_TOO_SMALL;
244 }
245 }
246
247 if (Status != EFI_BUFFER_TOO_SMALL) {
248 Status = gRT->SetVariable (
249 UnicodeVariableName,
250 VariableGuid,
251 Attributes,
252 VariableSize,
253 VariableData
254 );
255 DEBUG ((
256 EFI_ERROR (Status) && VariableSize > 0 ? DEBUG_WARN : DEBUG_INFO,
257 "OC: Setting NVRAM %g:%a - %r\n",
258 VariableGuid,
259 AsciiVariableName,
260 Status
261 ));
262 } else {
263 DEBUG ((
264 DEBUG_INFO,
265 "OC: Setting NVRAM %g:%a - ignored, exists\n",
266 VariableGuid,
267 AsciiVariableName,
268 Status
269 ));
270 }
271
272 FreePool (UnicodeVariableName);
273}

277OcLoadLegacyNvram (
278 IN EFI_SIMPLE_FILE_SYSTEM_PROTOCOL *FileSystem,
279 IN OC_GLOBAL_CONFIG *Config
280 )
281{
282 UINT8 *FileBuffer;
283 UINT32 FileSize;
284 OC_NVRAM_STORAGE Nvram;
285 BOOLEAN IsValid;
286 EFI_STATUS Status;
287 UINT32 GuidIndex;
288 UINT32 VariableIndex;
289 GUID VariableGuid;
290 OC_ASSOC *VariableMap;
291 OC_NVRAM_LEGACY_ENTRY *SchemaEntry;
292 OC_NVRAM_LEGACY_MAP *Schema;
293
294 Schema = &Config->Nvram.Legacy;
295
296 FileBuffer = OcReadFile (FileSystem, OPEN_CORE_NVRAM_PATH, &FileSize, BASE_1MB);
297 if (FileBuffer == NULL) {
298 DEBUG ((DEBUG_INFO, "OC: Invalid nvram data\n"));
299 return;
300 }
301
302 OC_NVRAM_STORAGE_CONSTRUCT (&Nvram, sizeof (Nvram));
303 IsValid = ParseSerialized (&Nvram, &mNvramStorageRootSchema, FileBuffer, FileSize, NULL);
304 FreePool (FileBuffer);
305
306 if (!IsValid || (Nvram.Version != OC_NVRAM_STORAGE_VERSION)) {
307 DEBUG ((
308 DEBUG_WARN,
309 "OC: Incompatible nvram data, version %u vs %d\n",
310 Nvram.Version,
311 OC_NVRAM_STORAGE_VERSION
312 ));
313 OC_NVRAM_STORAGE_DESTRUCT (&Nvram, sizeof (Nvram));
314 return;
315 }
316
317 for (GuidIndex = 0; GuidIndex < Nvram.Add.Count; ++GuidIndex) {
318 Status = OcProcessVariableGuid (
319 OC_BLOB_GET (Nvram.Add.Keys[GuidIndex]),
320 &VariableGuid,
321 Schema,
322 &SchemaEntry
323 );
324
325 if (EFI_ERROR (Status)) {
326 continue;
327 }
328
329 VariableMap = Nvram.Add.Values[GuidIndex];
330
331 for (VariableIndex = 0; VariableIndex < VariableMap->Count; ++VariableIndex) {
332 OcSetNvramVariable (
333 OC_BLOB_GET (VariableMap->Keys[VariableIndex]),
334 &VariableGuid,
335 Config->Nvram.WriteFlash ? OPEN_CORE_NVRAM_NV_ATTR : OPEN_CORE_NVRAM_ATTR,
336 VariableMap->Values[VariableIndex]->Size,
337 OC_BLOB_GET (VariableMap->Values[VariableIndex]),
338 SchemaEntry,
339 Config->Nvram.LegacyOverwrite
340 );
341 }
342 }
343
344 OC_NVRAM_STORAGE_DESTRUCT (&Nvram, sizeof (Nvram));

348VOID
349OcDeleteNvram (
350 IN OC_GLOBAL_CONFIG *Config
351 )
352{
353 EFI_STATUS Status;
354 UINT32 DeleteGuidIndex;
355 UINT32 AddGuidIndex;
356 UINT32 DeleteVariableIndex;
357 UINT32 AddVariableIndex;
358 CONST CHAR8 *AsciiVariableName;
359 CHAR16 *UnicodeVariableName;
360 GUID VariableGuid;
361 OC_ASSOC *VariableMap;
362 VOID *CurrentValue;
363 UINTN CurrentValueSize;
364 BOOLEAN SameContents;
365
366 for (DeleteGuidIndex = 0; DeleteGuidIndex < Config->Nvram.Delete.Count; ++DeleteGuidIndex) {
367 Status = OcProcessVariableGuid (
368 OC_BLOB_GET (Config->Nvram.Delete.Keys[DeleteGuidIndex]),
369 &VariableGuid,
370 NULL,
371 NULL
372 );
373
374 if (EFI_ERROR (Status)) {
375 continue;
376 }
377
378 //
379 // When variable is set and non-volatile variable setting is used,
380 // we do not want a variable to be constantly removed and added every reboot,
381 // as it will negatively impact flash memory. In case the variable is already set
382 // and has the same value we do not delete it.
383 //
384 for (AddGuidIndex = 0; AddGuidIndex < Config->Nvram.Add.Count; ++AddGuidIndex) {
385 if (AsciiStrCmp (
386 OC_BLOB_GET (Config->Nvram.Delete.Keys[DeleteGuidIndex]),
387 OC_BLOB_GET (Config->Nvram.Add.Keys[AddGuidIndex])
388 ) == 0)
389 {
390 break;
391 }
392 }
393
394 for (DeleteVariableIndex = 0; DeleteVariableIndex < Config->Nvram.Delete.Values[DeleteGuidIndex]->Count; ++DeleteVariableIndex) {
395 AsciiVariableName = OC_BLOB_GET (Config->Nvram.Delete.Values[DeleteGuidIndex]->Values[DeleteVariableIndex]);
396
397 //
398 // '#' is filtered in all keys, but for values we need to do it ourselves.
399 //
400 if (AsciiVariableName[0] == '#') {
401 DEBUG ((DEBUG_INFO, "OC: Variable skip deleting %a\n", AsciiVariableName));
402 continue;
403 }
404
405 UnicodeVariableName = AsciiStrCopyToUnicode (AsciiVariableName, 0);
406
407 if (UnicodeVariableName == NULL) {
408 DEBUG ((DEBUG_WARN, "OC: Failed to convert NVRAM variable name %a\n", AsciiVariableName));
409 continue;
410 }
411
412 if (AddGuidIndex != Config->Nvram.Add.Count) {
413 VariableMap = NULL;
414 for (AddVariableIndex = 0; AddVariableIndex < Config->Nvram.Add.Values[AddGuidIndex]->Count; ++AddVariableIndex) {
415 if (AsciiStrCmp (AsciiVariableName, OC_BLOB_GET (Config->Nvram.Add.Values[AddGuidIndex]->Keys[AddVariableIndex])) == 0) {
416 VariableMap = Config->Nvram.Add.Values[AddGuidIndex];
417 break;
418 }
419 }
420
421 if (VariableMap != NULL) {
422 Status = GetVariable2 (UnicodeVariableName, &VariableGuid, &CurrentValue, &CurrentValueSize);
423
424 if (!EFI_ERROR (Status)) {
425 SameContents = CurrentValueSize == VariableMap->Values[AddVariableIndex]->Size
426 && CompareMem (OC_BLOB_GET (VariableMap->Values[AddVariableIndex]), CurrentValue, CurrentValueSize) == 0;
427 FreePool (CurrentValue);
428 } else if ((Status == EFI_NOT_FOUND) && (VariableMap->Values[AddVariableIndex]->Size == 0)) {
429 SameContents = TRUE;
430 } else {
431 SameContents = FALSE;
432 }
433
434 if (SameContents) {
435 DEBUG ((DEBUG_INFO, "OC: Not deleting NVRAM %g:%a, matches add\n", &VariableGuid, AsciiVariableName));
436 FreePool (UnicodeVariableName);
437 continue;
438 }
439 }
440 }
441
442 Status = gRT->SetVariable (UnicodeVariableName, &VariableGuid, 0, 0, 0);
443 DEBUG ((
444 EFI_ERROR (Status) && Status != EFI_NOT_FOUND ? DEBUG_WARN : DEBUG_INFO,
445 "OC: Deleting NVRAM %g:%a - %r\n",
446 &VariableGuid,
447 AsciiVariableName,
448 Status
449 ));
450
451 FreePool (UnicodeVariableName);
452 }
453 }
