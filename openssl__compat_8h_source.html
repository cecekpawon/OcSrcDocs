68static inline int
69RSA_set0_key(RSA *rsa, BIGNUM *n, BIGNUM *e, BIGNUM *d)
70{
71 if ((rsa->n == NULL && n == NULL)
72 || (rsa->e == NULL && e == NULL))
73 {
74 return 0;
75 }
76
77 if (n != NULL)
78 {
79 BN_free(rsa->n);
80 rsa->n = n;
81 }
82 if (e != NULL)
83 {
84 BN_free(rsa->e);
85 rsa->e = e;
86 }
87 if (d != NULL)
88 {
89 BN_free(rsa->d);

91 }
92
93 return 1;
94}
