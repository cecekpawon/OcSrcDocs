229// Input/Output/Bidirectional Stream Descriptor n Format; 2 bytes.
230#define HDA_REG_SDNFMT(n) (0x92 + (0x20 * (n)))
231#define HDA_REG_SDNFMT_CHAN(a) ((UINT8)((a) & 0xF))
232#define HDA_REG_SDNFMT_BITS(a) ((UINT8)(((a) >> 4) & 0x3))
233#define HDA_REG_SDNFMT_BITS_8 0x0
234#define HDA_REG_SDNFMT_BITS_16 0x1
235#define HDA_REG_SDNFMT_BITS_20 0x2
236#define HDA_REG_SDNFMT_BITS_24 0x3
237#define HDA_REG_SDNFMT_BITS_32 0x4
238#define HDA_REG_SDNFMT_DIV(a) ((UINT8)(((a) >> 8) & 0x3))
239#define HDA_REG_SDNFMT_MULT(a) ((UINT8)(((a) >> 11) & 0x3))
240#define HDA_REG_SDNFMT_BASE_44KHZ BIT14
241#define HDA_REG_SDNFMT_SET(chan, bits, div, mult, base) \
242 ((UINT16)(((chan) & 0xF) | (((bits) & 0x3) << 4) | (((div) & 0x3) << 8) | \
243 (((mult) & 0x3) << 11) | ((base) ? HDA_REG_SDNFMT_BASE_44KHZ : 0)))
