23STATIC
24UINT16
25ThumbMovtImmediateAddress (
26 IN CONST VOID *Instruction
27 )
28{
29 UINT32 Movt;
30 UINT16 Movt1;
31 UINT16 Movt2;
32 UINT16 Address;
33
34 //
35 // Thumb2 is two separate 16-bit instructions working together, e.g.
36 // MOVT R0, #0 is 0x0000f2c0 or 0xf2c0 0x0000
37 //
38 Movt1 = READ_ALIGNED_16 (Instruction);
39 Movt2 = READ_ALIGNED_16 ((CONST CHAR8 *)Instruction + sizeof (UINT16));
40 Movt = COMPOSE_32 (Movt1, Movt2);
41 //
42 // imm16 = imm4:i:imm3:imm8
43 // imm4 -> Bit19:Bit16
44 // i -> Bit26
45 // imm3 -> Bit14:Bit12
46 // imm8 -> Bit7:Bit0
47 //
48 Address = (UINT16)(Movt & 0x000000FFU); // imm8
49 Address |= (UINT16)((Movt >> 4U) & 0x0000F700U); // imm4 imm3
50 Address |= (UINT16)((Movt & BIT26) >> 15U); // i, Bit26->11
51 return Address;
52}

61STATIC
62VOID
63ThumbMovtImmediatePatch (
64 IN OUT VOID *Instruction,
65 IN UINT16 Address
66 )
67{
68 UINT16 Patch;
69 UINT16 PatchedInstruction;
70
71 //
72 // First 16-bit chunk of instruction.
73 //
74 Patch = (Address & 0xF000U) >> 12U; // imm4
75 Patch |= (Address & BIT11) >> 1U; // i, Bit11->10
76 //
77 // Mask out instruction bits and or in address.
78 //
79 PatchedInstruction = READ_ALIGNED_16 (Instruction);
80 WRITE_ALIGNED_16 (Instruction, (PatchedInstruction & ~(UINT16)0x040FU) | Patch);
81 //
82 // Second 16-bit chunk of instruction.
83 //
84 Patch = Address & 0x000000FFU; // imm8
85 Patch |= ((UINT32)Address << 4U) & 0x00007000U; // imm3
86 //
87 // Mask out instruction bits and or in address.
88 //
89 PatchedInstruction = READ_ALIGNED_16 ((CHAR8 *)Instruction + sizeof (UINT16));
90 WRITE_ALIGNED_16 (
91 (CHAR8 *)Instruction + sizeof (UINT16),
92 (PatchedInstruction & ~(UINT16)0x70FFU) | Patch
93 );
94}

373 ASSERT (RelocationData != NULL || RelocationDataSize == 0);
374 ASSERT (RelocationData == NULL || RelocationDataSize >= sizeof (PE_COFF_RUNTIME_CONTEXT) + Context->RelocDirSize * (sizeof (UINT64) / sizeof (UINT16)));
375 //
376 // Calculate the Image displacement from its prefered load address.
377 //
378
379 Adjust = BaseAddress - Context->ImageBase;
380 //
381 // Runtime drivers should unconditionally go through the full Relocation
382 // procedure early to eliminate the possibility of errors later at runtime.
383 // Runtime drivers don't have their Base Relocations stripped, this is
384 // verified during context creation.
385 // Skip explicit Relocation when the Image is already loaded at its
386 // prefered location.
387 //
388
389 if ((RelocationData == NULL) && (Adjust == 0)) {
390 return RETURN_SUCCESS;
391 }
392
393 if (RelocationData != NULL) {
394 RelocationData->RelocDirRva = Context->RelocDirRva;
395 RelocationData->RelocDirSize = Context->RelocDirSize;
396 }
397
398 //
399 // Apply Base Relocation fixups to the image.
400 //
401
402 RelocOffset = Context->RelocDirRva;
403 RelocMax = Context->RelocDirRva + Context->RelocDirSize - sizeof (EFI_IMAGE_BASE_RELOCATION_BLOCK);
404 RelocDataIndex = 0;
405
406 //
407 // Process all Base Relocation Blocks.
408 //
409
410 while (RelocOffset <= RelocMax) {
411 RelocWalker = (CONST EFI_IMAGE_BASE_RELOCATION_BLOCK *)(CONST VOID *)(
412 (CONST CHAR8 *)Context->ImageBuffer + RelocOffset
413 );
414
415 Result = BaseOverflowSubU32 (
416 RelocWalker->SizeOfBlock,
417 sizeof (EFI_IMAGE_BASE_RELOCATION_BLOCK),
418 &SizeOfRelocs
419 );
420 //
421 // Ensure there is at least one entry.
422 // Ensure the block's size is padded to ensure proper alignment.
423 //
424 if (Result) {
425 return RETURN_UNSUPPORTED;
426 }
427
428 if (SizeOfRelocs > RelocMax - RelocOffset) {
429 return RETURN_UNSUPPORTED;
430 }
431
432 if (!IS_ALIGNED (RelocWalker->SizeOfBlock, OC_ALIGNOF (EFI_IMAGE_BASE_RELOCATION_BLOCK))) {
433 return RETURN_UNSUPPORTED;
434 }
435
436 //
437 // This division is safe due to the guarantee made above.
438 //
439
440 NumRelocs = SizeOfRelocs / sizeof (*RelocWalker->Relocations);
441 //
442 // Apply all Base Relocation fixups of the current block.
443 //
444 if (RelocationData != NULL) {
445 WalkerFixupData = &RelocationData->FixupData[RelocDataIndex];
446 } else {
447 WalkerFixupData = NULL;
448 }
449
450 //
451 // Process all Base Relocations of the current Block.
452 //
453 for (RelocIndex = 0; RelocIndex < NumRelocs; ++RelocIndex) {
454 //
455 // Apply the Base Relocation fixup per type.
456 // If RelocationData is not NULL, store the current value of the fixup
457 // target to determine whether it has been changed during runtime
458 // execution.
459 //
460 // It is not clear how EFI_IMAGE_REL_BASED_HIGH and
461 // EFI_IMAGE_REL_BASED_LOW are supposed to be handled. While PE reference
462 // suggests to just add the high or low part of the displacement, there
463 // are concerns about how it's supposed to deal with wraparounds.
464 // As neither LLD,
465 //
466
467 Status = InternalApplyRelocation (
468 Context,
469 RelocWalker,
470 RelocIndex,
471 Adjust,
472 WalkerFixupData
473 );
474
475 if (Status != RETURN_SUCCESS) {
476 return Status;
477 }
478 }
479
480 RelocDataIndex += NumRelocs;
481 RelocOffset += RelocWalker->SizeOfBlock;
482 }
483
484 //
485 // Ensure the Relocation Directory size matches the contained data.
486 //
487
488 if (RelocOffset != RelocMax + sizeof (EFI_IMAGE_BASE_RELOCATION_BLOCK)) {
489 return RETURN_UNSUPPORTED;
490 }
491
492 //
493 // Initialise the still uninitialised portion of the Runtime context.
494 //
495 if (RelocationData != NULL) {
496 ZeroMem (
497 &RelocationData->FixupData[RelocDataIndex],
498 RelocationDataSize - sizeof (PE_COFF_RUNTIME_CONTEXT) - RelocDataIndex * sizeof (UINT64)
499 );
500 }
501
502 return RETURN_SUCCESS;
503}

523InternalApplyRelocationRuntime (
524 IN OUT VOID *Fixup,
525 IN UINT16 RelocType,
526 IN UINT64 Adjust,
527 IN UINT64 FixupData
528 )
529{
530 UINT32 Fixup32;
531 UINT64 Fixup64;
532
533 ASSERT (Fixup != NULL);
534 ASSERT (
535 IMAGE_RELOC_TYPE_SUPPORTED (RelocType)
536 && RelocType != EFI_IMAGE_REL_BASED_ABSOLUTE

540 /* LCOV_EXCL_BR_LINE */

553 break;
554
555 case EFI_IMAGE_REL_BASED_DIR64:

557 if (FixupData != Fixup64) {
558 if (PcdGetBool (PcdImageLoaderRtRelocAllowTargetMismatch)) {
559 return RETURN_SUCCESS;
560 }
561
562 return RETURN_UNSUPPORTED;
563 }
564
565 Fixup64 += Adjust;
566 WriteUnaligned64 (Fixup, Fixup64);
567 break;
568
569 case EFI_IMAGE_REL_BASED_ARM_MOV32T:
570 ASSERT (PcdGetBool (PcdImageLoaderSupportArmThumb));
571 Fixup64 = ReadUnaligned64 (Fixup);
572 if (FixupData != Fixup64) {
573 if (PcdGetBool (PcdImageLoaderRtRelocAllowTargetMismatch)) {
574 return RETURN_SUCCESS;
575 }
576
577 return RETURN_UNSUPPORTED;
578 }
579
580 ThumbMovwMovtImmediateFixup (Fixup, Adjust);
581 break;
582
583 /* LCOV_EXCL_START */
584 default:
585 ASSERT (FALSE);
586 }
587
588 /* LCOV_EXCL_STOP */
589
590 return RETURN_SUCCESS;
591}

637PeCoffRelocateImageForRuntime (
638 IN OUT VOID *Image,
639 IN UINT32 ImageSize,
640 IN UINT64 BaseAddress,
641 IN CONST PE_COFF_RUNTIME_CONTEXT *RelocationData
642 )
643{
644 UINTN ImageAddress;
645 UINT32 FixupDataIndex;
646 CONST EFI_IMAGE_BASE_RELOCATION_BLOCK *RelocWalker;
647 UINT64 Adjust;
648 RETURN_STATUS Status;
649
650 UINT32 RelocOffset;
651 UINT32 SizeOfRelocs;
652 UINT32 NumRelocs;
653 UINT32 RelocIndex;
654 UINT32 RelocTarget;
655 UINT32 RelocSuboffset;
656
657 (VOID)ImageSize;
658
659 ASSERT (Image != NULL);
660 ASSERT (BaseAddress != 0);
661 ASSERT (RelocationData != NULL);
662 //
663 // This function assumes the image has previously been validated by
664 // PeCoffInitializeContext().
665 //
666
667 ImageAddress = PTR_TO_ADDR (Image, 0);
668
669 Adjust = BaseAddress - ImageAddress;
670 if (Adjust == 0) {
671 return RETURN_SUCCESS;
672 }
673
674 FixupDataIndex = 0;
675 RelocOffset = RelocationData->RelocDirRva;
676
677 while (RelocOffset < RelocationData->RelocDirRva + RelocationData->RelocDirSize) {
678 RelocWalker = (CONST EFI_IMAGE_BASE_RELOCATION_BLOCK *)(CONST VOID *)(
679 (CONST CHAR8 *)Image + RelocOffset
680 );
681
682 STATIC_ASSERT (
683 (sizeof (UINT32) % OC_ALIGNOF (EFI_IMAGE_BASE_RELOCATION_BLOCK)) == 0,
684 "The following accesses must be performed unaligned."
685 );
686
687 ASSERT ((UINT32)sizeof (EFI_IMAGE_BASE_RELOCATION_BLOCK) <= (MAX_UINT32 - RelocWalker->SizeOfBlock));
688
689 SizeOfRelocs = RelocWalker->SizeOfBlock - sizeof (EFI_IMAGE_BASE_RELOCATION_BLOCK);
690

694 //

696

699 //

708 //

713

716 IMAGE_RELOC_TYPE (RelocWalker->Relocations[RelocIndex]),
717 Adjust,
718 RelocationData->FixupData[FixupDataIndex]
719 );
720
721 if (!PcdGetBool (PcdImageLoaderRtRelocAllowTargetMismatch) && (Status != RETURN_SUCCESS)) {
722 return Status;
723 }

726 }
727

729 }

731 ASSERT (RelocOffset == RelocationData->RelocDirRva + RelocationData->RelocDirSize);
