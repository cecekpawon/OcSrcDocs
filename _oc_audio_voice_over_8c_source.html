139 CONST CHAR8 *Language OPTIONAL
140 )
141{
142 EFI_STATUS Status;
143 UINTN LanguageSize;
144 CHAR8 LanguageData[16];
145 APPLE_VOICE_OVER_AUDIO_PROTOCOL *VoiceOver;
146
147 if (Language == NULL) {
148 LanguageSize = sizeof (LanguageData) - 1;
149 Status = gRT->GetVariable (
150 APPLE_PREV_LANG_KBD_VARIABLE_NAME,
151 &gAppleBootVariableGuid,
152 NULL,
153 &LanguageSize,
154 &LanguageData[0]
155 );
156 if (!EFI_ERROR (Status)) {
157 LanguageData[LanguageSize] = 0;
158 Language = LanguageData;
159 }
160 } else {
161 Status = EFI_SUCCESS;
162 }
163
164 if (!EFI_ERROR (Status)) {
165 Status = gBS->LocateProtocol (
166 &gAppleVOAudioProtocolGuid,
167 NULL,
168 (VOID **)&VoiceOver
169 );
170
171 if (!EFI_ERROR (Status)) {
172 Status = VoiceOver->SetLanguageString (VoiceOver, Language);
173 }
174
175 DEBUG ((DEBUG_INFO, "OCAU: Language for audio %a - %r\n", Language, Status));
176 } else {
177 DEBUG ((DEBUG_INFO, "OCAU: No language for audio - %r\n", Status));
178 }
179
180 return Status;
181}

184EFIAPI
185InternalOcAudioVoiceOverPlay (
186 IN APPLE_VOICE_OVER_AUDIO_PROTOCOL *This,
187 IN UINT8 File
188 )
189{
190 OC_AUDIO_PROTOCOL_PRIVATE *Private;
191 CONST APPLE_VOICE_OVER_FILE_MAP *Map;
192 BOOLEAN Localised;
193
194 Localised = FALSE;
195 Map = NULL;
196
197 if ( (File >= AppleVoiceOverAudioFileIndexLocalisedMin)
198 && (File < AppleVoiceOverAudioFileIndexLocalisedMax))
199 {
200 Localised = TRUE;
201 Map = &mAppleVoiceOverLocalisedAudioFiles[File - AppleVoiceOverAudioFileIndexLocalisedMin];
202 } else if ( (File >= AppleVoiceOverAudioFileIndexNonLocalisedMin)
203 && (File < AppleVoiceOverAudioFileIndexNonLocalisedMax))
204 {
205 Map = &mAppleVoiceOverNonLocalisedAudioFiles[File - AppleVoiceOverAudioFileIndexNonLocalisedMin];
206 }
207
208 if ((Map == NULL) || (Map->BasePath == NULL) || (Map->BaseType == NULL)) {
209 DEBUG ((DEBUG_INFO, "OCAU: Unsupported Apple voice over file index %u\n", File));
210 return EFI_UNSUPPORTED;
211 }
212
213 Private = OC_AUDIO_PROTOCOL_PRIVATE_FROM_VOICE_OVER (This);
214 return Private->OcAudio.PlayFile (&Private->OcAudio, Map->BasePath, Map->BaseType, Localised, 0, FALSE, TRUE);
215}

236EFIAPI
237InternalOcAudioVoiceOverSetLanguageString (
238 IN APPLE_VOICE_OVER_AUDIO_PROTOCOL *This,
239 IN CONST CHAR8 *LanguageString
240 )
241{
242 OC_AUDIO_PROTOCOL_PRIVATE *Private;
243 UINTN Index;
244
245 DEBUG ((DEBUG_INFO, "OCAU: Setting language to %a\n", LanguageString));
246
247 Private = OC_AUDIO_PROTOCOL_PRIVATE_FROM_VOICE_OVER (This);
248
249 for (Index = 1; Index < ARRAY_SIZE (mLanguagePairing); ++Index) {
250 if (AsciiStrCmp (mLanguagePairing[Index], LanguageString) == 0) {
251 Private->Language = (UINT8)Index;
252 return EFI_SUCCESS;
253 }
254 }
255
256 for (Index = 1; Index < ARRAY_SIZE (mLanguagePairing); ++Index) {
257 if ( (AsciiStrLen (mLanguagePairing[Index]) == 2)
258 && (AsciiStrnCmp (mLanguagePairing[Index], LanguageString, 2) == 0))
259 {
260 Private->Language = (UINT8)Index;
261 return EFI_SUCCESS;
262 }
263 }
264
265 if (AsciiStrnCmp (LanguageString, "zh", 2) == 0) {
266 Private->Language = AppleVoiceOverLanguageZhHans;
267 return EFI_SUCCESS;
268 }
269
270 return EFI_INVALID_PARAMETER;
271}
