132 if (!mGUIDMaps[Index].NvramKeyMaps[Index2].KeyChecker (

134 VariableMap->Values[VariableIndex]->Size
135 ))

137 DEBUG ((
138 DEBUG_WARN,
139 "NVRAM->Add->%g->%a has illegal value!\n",
140 &Guid,
141 OC_BLOB_GET (VariableMap->Keys[VariableIndex])
142 ));
143 ++ErrorCount;
144 } else {
145 //
