49 //

61 );
62 //
63 // 10. Using the PointerToRawData field (offset 20) in the referenced
64 // SectionHeader structure as a key, arrange the table's elements in
65 // ascending order. In other words, sort the section headers in ascending
66 // order according to the disk-file offset of the sections.
67 //
68
69 SortedSections[0] = &Sections[0];

71 // Perform Insertion Sort.
72 //
73 for (SectIndex = 1; SectIndex < Context->NumberOfSections; ++SectIndex) {
74 for ( SectionPos = SectIndex;
75 0 < SectionPos
76 && SortedSections[SectionPos - 1]->PointerToRawData > Sections[SectIndex].PointerToRawData;

82 SortedSections[SectionPos] = &Sections[SectIndex];
83 }
84
85 Result = TRUE;
86 SectionTop = 0;
87
88 //
89 // 13. Repeat steps 11 and 12 for all of the sections in the sorted table.
90 //
91 for (SectIndex = 0; SectIndex < Context->NumberOfSections; ++SectIndex) {
92 if (PcdGetBool (PcdImageLoaderHashProhibitOverlap)) {
93 if (SectionTop > SortedSections[SectIndex]->PointerToRawData) {
94 Result = FALSE;
95 break;
96 }
97
98 SectionTop = SortedSections[SectIndex]->PointerToRawData + SortedSections[SectIndex]->SizeOfRawData;
99 }
100
101 //
102 // 11. Walk through the sorted table, load the corresponding section into
103 // memory, and hash the entire section. Use the SizeOfRawData field in the
104 // SectionHeader structure to determine the amount of data to hash.
105 //
106
107 if (SortedSections[SectIndex]->SizeOfRawData > 0) {
108 Result = HashUpdate (
109 HashContext,
110 (CONST CHAR8 *)Context->FileBuffer + SortedSections[SectIndex]->PointerToRawData,
111 SortedSections[SectIndex]->SizeOfRawData
112 );

115 }
116 }
117 }
