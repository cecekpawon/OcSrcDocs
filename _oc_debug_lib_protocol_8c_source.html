66LogProtocolArrivedNotify (

70{
71 UINTN Index;
72 EFI_STATUS Status;
73 OC_LOG_PROTOCOL *OcLog;
74 CHAR8 *Walker;
75 UINTN ErrorLevel;
76 OC_LOG_OPTIONS CurrLogOpt;
77
78 //
79 // Event arrives. Close it.
80 //
81 gBS->CloseEvent (mLogProtocolArrivedNotifyEvent);
82
83 Status = gBS->LocateProtocol (
84 &gOcLogProtocolGuid,
85 NULL,
86 (VOID **)&OcLog
87 );
88
89 if (EFI_ERROR (Status) || (OcLog->Revision != OC_LOG_REVISION)) {
90 return;
91 }
92
93 Walker = (CHAR8 *)mLogBuffer;
94 for (Index = 0; Index < mLogCount; ++Index) {
95 //
96 // Set ErrorLevel.
97 //
98 CopyMem (&ErrorLevel, Walker, sizeof (ErrorLevel));
99 Walker += sizeof (ErrorLevel);
100
101 //
102 // Print debug message without onscreen, as it is done by OutputString.
103 //
104 CurrLogOpt = OcLog->Options;
105 OcLog->Options &= ~OC_LOG_CONSOLE;
106 DebugPrint (ErrorLevel, "%a", Walker);
107 //
108 // Restore original value.
109 //
110 OcLog->Options = CurrLogOpt;
111
112 //
113 // Skip message chars.
114 //
115 while (*Walker != '\0') {
116 ++Walker;
117 }
118
119 //
120 // Matched '\0'. Go to the next message.
121 //
122 ++Walker;
123 }
124
125 FreePool (mLogBuffer);
126 mLogBuffer = NULL;

131OcBufferEarlyLog (
132 IN UINTN ErrorLevel,
133 IN CONST CHAR16 *Buffer
134 )
135{
136 EFI_STATUS Status;
137
138 if (mLogBuffer == NULL) {
139 mLogBuffer = AllocatePool (OC_LOG_BUFFER_SIZE);
140 if (mLogBuffer == NULL) {
141 return;
142 }
143
144 mLogWalker = mLogBuffer;
145
146 //
147 // Notify when log protocol arrives.
148 //
149 Status = gBS->CreateEvent (
150 EVT_NOTIFY_SIGNAL,
151 TPL_NOTIFY,
152 LogProtocolArrivedNotify,
153 NULL,
154 &mLogProtocolArrivedNotifyEvent
155 );
156
157 if (!EFI_ERROR (Status)) {
158 Status = gBS->RegisterProtocolNotify (
159 &gOcLogProtocolGuid,
160 mLogProtocolArrivedNotifyEvent,
161 &mLogProtocolArrivedNotifyRegistration
162 );
163
164 if (EFI_ERROR (Status)) {
165 gBS->CloseEvent (mLogProtocolArrivedNotifyEvent);
166 }
167 }
168 }
169
170 //
171 // The size of ErrorLevel and that of Buffer string including null terminator should not overflow.
172 // The current position (mLogWalker) has been moved to the right side for the same purpose.
173 //
174 if ((sizeof (ErrorLevel) + StrLen (Buffer) + 1) <= (UINTN)(mLogBuffer + OC_LOG_BUFFER_SIZE - mLogWalker)) {
175 //
176 // Store ErrorLevel into buffer.
177 //
178 CopyMem (mLogWalker, &ErrorLevel, sizeof (ErrorLevel));
179 mLogWalker += sizeof (ErrorLevel);
180
181 //
182 // Store logs into buffer.
183 //
184 while (*Buffer != CHAR_NULL) {
185 *mLogWalker++ = (UINT8)*Buffer++;
186 }
187
188 *mLogWalker++ = CHAR_NULL;
189
190 //
191 // Increment number of messages.
192 //
193 ++mLogCount;
194 }
195}
