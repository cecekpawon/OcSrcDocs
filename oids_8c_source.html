454bool DEROidCompare(const DERItem *oid1, const DERItem *oid2) {
455 if ((oid1 == NULL) || (oid2 == NULL)) {
456 return false;
457 }
458 if (oid1->length != oid2->length) {
459 return false;
460 }
461 if (!DERMemcmp(oid1->data, oid2->data, oid1->length)) {
462 return true;
463 } else {
464 return false;
465 }
466}
