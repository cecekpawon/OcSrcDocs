249EFIAPI
250OcImageLoaderLoad (
251 IN BOOLEAN BootPolicy,
252 IN EFI_HANDLE ParentImageHandle,
253 IN EFI_DEVICE_PATH_PROTOCOL *DevicePath,
254 IN VOID *SourceBuffer OPTIONAL,
255 IN UINTN SourceSize,
256 OUT EFI_HANDLE *ImageHandle
257 )
258{
259 EFI_STATUS Status;
260 EFI_STATUS ImageStatus;
261 PE_COFF_IMAGE_CONTEXT ImageContext;
262 EFI_PHYSICAL_ADDRESS DestinationArea;
263 UINT32 DestinationSize;
264 VOID *DestinationBuffer;
265 OC_LOADED_IMAGE_PROTOCOL *OcLoadedImage;
266 EFI_LOADED_IMAGE_PROTOCOL *LoadedImage;
267
268 ASSERT (SourceBuffer != NULL);
269
270 //
271 // Reject very large files.
272 //
273 if (SourceSize > MAX_UINT32) {
274 return EFI_UNSUPPORTED;
275 }
276
277 //
278 // Initialize the image context.
279 //
280 ImageStatus = PeCoffInitializeContext (
281 &ImageContext,
282 SourceBuffer,
283 (UINT32)SourceSize
284 );
285 if (EFI_ERROR (ImageStatus)) {
286 DEBUG ((DEBUG_INFO, "OCB: PeCoff init failure - %r\n", ImageStatus));
287 return EFI_UNSUPPORTED;
288 }
289
290 //
291 // Reject images that are not meant for the platform's architecture.
292 //
293 if (ImageContext.Machine != OC_IMAGE_FILE_MACHINE) {
294 DEBUG ((DEBUG_INFO, "OCB: PeCoff wrong machine - %x\n", ImageContext.Machine));
295 return EFI_UNSUPPORTED;
296 }
297
298 //
299 // Reject RT drivers for the moment.
300 //
301 if (ImageContext.Subsystem == EFI_IMAGE_SUBSYSTEM_EFI_RUNTIME_DRIVER) {
302 DEBUG ((DEBUG_INFO, "OCB: PeCoff no support for RT drivers\n"));
303 return EFI_UNSUPPORTED;
304 }
305
306 //
307 // FIXME: This needs to be backported as a function:
308 // https://github.com/mhaeuser/edk2/blob/2021-gsoc-secure-loader/MdePkg/Library/BaseUefiImageLib/CommonSupport.c#L19-L53
309 //
310 DestinationSize = ImageContext.SizeOfImage + ImageContext.SizeOfImageDebugAdd;
311 if (OcOverflowAddU32 (DestinationSize, ImageContext.SectionAlignment, &DestinationSize)) {
312 return RETURN_UNSUPPORTED;
313 }
314
315 if (DestinationSize >= BASE_16MB) {
316 DEBUG ((DEBUG_INFO, "OCB: PeCoff prohibits files over 16M (%u)\n", DestinationSize));
317 return RETURN_UNSUPPORTED;
318 }
319
320 //
321 // Allocate the image destination memory.
322 // FIXME: RT drivers require EfiRuntimeServicesCode.
323 //
324 Status = gBS->AllocatePages (
325 AllocateAnyPages,
326 ImageContext.Subsystem == EFI_IMAGE_SUBSYSTEM_EFI_APPLICATION
327 ? EfiLoaderCode : EfiBootServicesCode,
328 EFI_SIZE_TO_PAGES (ImageContext.SizeOfImage),
329 &DestinationArea
330 );
331 if (EFI_ERROR (Status)) {
332 return Status;
333 }
334
335 DestinationBuffer = (VOID *)(UINTN)DestinationArea;
336
337 //
338 // Load SourceBuffer into DestinationBuffer.
339 //
340 ImageStatus = PeCoffLoadImage (
341 &ImageContext,
342 DestinationBuffer,
343 ImageContext.SizeOfImage
344 );
345 if (EFI_ERROR (ImageStatus)) {
346 DEBUG ((DEBUG_INFO, "OCB: PeCoff load image error - %r\n", ImageStatus));
347 FreePages (DestinationBuffer, EFI_SIZE_TO_PAGES (ImageContext.SizeOfImage));
348 return EFI_UNSUPPORTED;
349 }
350
351 //
352 // Relocate the loaded image to the destination address.
353 //
354 ImageStatus = PeCoffRelocateImage (
355 &ImageContext,
356 (UINTN)DestinationBuffer,
357 NULL,
358 0
359 );
360 if (EFI_ERROR (ImageStatus)) {
361 DEBUG ((DEBUG_INFO, "OCB: PeCoff relocate image error - %r\n", ImageStatus));
362 FreePages (DestinationBuffer, EFI_SIZE_TO_PAGES (ImageContext.SizeOfImage));
363 return EFI_UNSUPPORTED;
364 }
365
366 //
367 // Construct a LoadedImage protocol for the image.
368 //
369 OcLoadedImage = AllocateZeroPool (sizeof (*OcLoadedImage));
370 if (OcLoadedImage == NULL) {
371 FreePages (DestinationBuffer, EFI_SIZE_TO_PAGES (ImageContext.SizeOfImage));
372 return EFI_OUT_OF_RESOURCES;
373 }
374
375 OcLoadedImage->EntryPoint = (EFI_IMAGE_ENTRY_POINT)((UINTN)DestinationBuffer + ImageContext.AddressOfEntryPoint);
376 OcLoadedImage->ImageArea = DestinationArea;
377 OcLoadedImage->PageCount = EFI_SIZE_TO_PAGES (ImageContext.SizeOfImage);
378 OcLoadedImage->Subsystem = ImageContext.Subsystem;
379
380 LoadedImage = &OcLoadedImage->LoadedImage;
381
382 LoadedImage->Revision = EFI_LOADED_IMAGE_INFORMATION_REVISION;
383 LoadedImage->ParentHandle = ParentImageHandle;
384 LoadedImage->SystemTable = gST;
385 LoadedImage->ImageBase = DestinationBuffer;
386 LoadedImage->ImageSize = ImageContext.SizeOfImage;
387 //
388 // FIXME: Support RT drivers.
389 //
390 if (ImageContext.Subsystem == EFI_IMAGE_SUBSYSTEM_EFI_APPLICATION) {
391 LoadedImage->ImageCodeType = EfiLoaderCode;
392 LoadedImage->ImageDataType = EfiLoaderData;
393 } else {
394 LoadedImage->ImageCodeType = EfiBootServicesCode;
395 LoadedImage->ImageDataType = EfiBootServicesData;
396 }
397
398 //
399 // Install LoadedImage and the image's entry point.
400 //
401 *ImageHandle = NULL;
402 Status = gBS->InstallMultipleProtocolInterfaces (
403 ImageHandle,
404 &gEfiLoadedImageProtocolGuid,
405 LoadedImage,
406 &mOcLoadedImageProtocolGuid,
407 OcLoadedImage,
408 NULL
409 );
410 if (EFI_ERROR (Status)) {
411 DEBUG ((DEBUG_INFO, "OCB: PeCoff proto install error - %r\n", Status));
412 FreePool (OcLoadedImage);
413 FreePages (DestinationBuffer, EFI_SIZE_TO_PAGES (ImageContext.SizeOfImage));
414 return Status;
415 }
416
417 DEBUG ((DEBUG_VERBOSE, "OCB: Loaded image at %p\n", *ImageHandle));
418
419 return EFI_SUCCESS;
420}

491STATIC
492EFI_STATUS
493InternalDirectExit (
494 IN OC_LOADED_IMAGE_PROTOCOL *OcLoadedImage,
495 IN EFI_HANDLE ImageHandle,
496 IN EFI_STATUS ExitStatus,
497 IN UINTN ExitDataSize,
498 IN CHAR16 *ExitData OPTIONAL
499 )
500{
501 EFI_TPL OldTpl;
502
503 DEBUG ((
504 DEBUG_VERBOSE,
505 "OCB: Exit %p %p (%d) - %r\n",
506 ImageHandle,
507 mCurrentImageHandle,
508 OcLoadedImage->Started,
509 ExitStatus
510 ));
511
512 //
513 // Prevent possible reentrance to this function for the same ImageHandle.
514 //
515 OldTpl = gBS->RaiseTPL (TPL_NOTIFY);
516
517 //
518 // If the image has not been started just free its resources.
519 // Should not happen normally.
520 //
521 if (!OcLoadedImage->Started) {
522 InternalDirectUnloadImage (OcLoadedImage, ImageHandle);
523 gBS->RestoreTPL (OldTpl);
524 return EFI_SUCCESS;
525 }
526
527 //
528 // If the image has been started, verify this image can exit.
529 //
530 if (ImageHandle != mCurrentImageHandle) {
531 DEBUG ((DEBUG_LOAD|DEBUG_ERROR, "OCB: Image is not exitable image\n"));
532 gBS->RestoreTPL (OldTpl);
533 return EFI_INVALID_PARAMETER;
534 }
535
536 //
537 // Set the return status.
538 //
539 OcLoadedImage->Status = ExitStatus;
540
541 //
542 // If there's ExitData info provide it.
543 //
544 if (ExitData != NULL) {
545 OcLoadedImage->ExitDataSize = ExitDataSize;
546 OcLoadedImage->ExitData = AllocatePool (OcLoadedImage->ExitDataSize);
547 if (OcLoadedImage->ExitData != NULL) {
548 CopyMem (OcLoadedImage->ExitData, ExitData, OcLoadedImage->ExitDataSize);
549 } else {
550 OcLoadedImage->ExitDataSize = 0;
551 }
552 }
553
554 //
555 // return to StartImage
556 //
557 gBS->RestoreTPL (OldTpl);
558 LongJump (OcLoadedImage->JumpContext, (UINTN)-1);
559
560 //
561 // If we return from LongJump, then it is an error
562 //
563 ASSERT (FALSE);
564 CpuDeadLoop ();
565 return EFI_ACCESS_DENIED;
566}

579STATIC
580EFI_STATUS
581InternalDirectStartImage (
582 IN OC_LOADED_IMAGE_PROTOCOL *OcLoadedImage,
583 IN EFI_HANDLE ImageHandle,
584 OUT UINTN *ExitDataSize,
585 OUT CHAR16 **ExitData OPTIONAL
586 )
587{
588 EFI_STATUS Status;
589 EFI_HANDLE LastImage;
590 UINTN SetJumpFlag;
591
592 //
593 // Push the current image.
594 //
595 LastImage = mCurrentImageHandle;
596 mCurrentImageHandle = ImageHandle;
597
598 //
599 // Set long jump for Exit() support
600 // JumpContext must be aligned on a CPU specific boundary.
601 // Overallocate the buffer and force the required alignment
602 //
603 OcLoadedImage->JumpBuffer = AllocatePool (
604 sizeof (BASE_LIBRARY_JUMP_BUFFER) + BASE_LIBRARY_JUMP_BUFFER_ALIGNMENT
605 );
606 if (OcLoadedImage->JumpBuffer == NULL) {
607 //
608 // Pop the current start image context
609 //
610 mCurrentImageHandle = LastImage;
611 return EFI_OUT_OF_RESOURCES;
612 }
613
614 OcLoadedImage->JumpContext = ALIGN_POINTER (
615 OcLoadedImage->JumpBuffer,
616 BASE_LIBRARY_JUMP_BUFFER_ALIGNMENT
617 );
618
619 SetJumpFlag = SetJump (OcLoadedImage->JumpContext);
620 //
621 // The initial call to SetJump() must always return 0.
622 // Subsequent calls to LongJump() cause a non-zero value to be returned by SetJump().
623 //
624 if (SetJumpFlag == 0) {
625 //
626 // Invoke the manually loaded image entry point.
627 //
628 DEBUG ((DEBUG_VERBOSE, "OCB: Starting image %p\n", ImageHandle));
629 OcLoadedImage->Started = TRUE;
630 OcLoadedImage->Status = OcLoadedImage->EntryPoint (
631 ImageHandle,
632 OcLoadedImage->LoadedImage.SystemTable
633 );
634 //
635 // If the image returns, exit it through Exit()
636 //
637 InternalDirectExit (OcLoadedImage, ImageHandle, OcLoadedImage->Status, 0, NULL);
638 }
639
640 FreePool (OcLoadedImage->JumpBuffer);
641
642 //
643 // Pop the current image.
644 //
645 mCurrentImageHandle = LastImage;
646
647 //
648 // NOTE: EFI 1.10 is not supported, refer to
649 // https://github.com/tianocore/edk2/blob/d8dd54f071cfd60a2dcf5426764a89cd91213420/MdeModulePkg/Core/Dxe/Image/Image.c#L1686-L1697
650 //
651
652 //
653 // Return the exit data to the caller
654 //
655 if ((ExitData != NULL) && (ExitDataSize != NULL)) {
656 *ExitDataSize = OcLoadedImage->ExitDataSize;
657 *ExitData = OcLoadedImage->ExitData;
658 } else if (OcLoadedImage->ExitData != NULL) {
659 //
660 // Caller doesn't want the exit data, free it
661 //
662 FreePool (OcLoadedImage->ExitData);
663 OcLoadedImage->ExitData = NULL;
664 }
665
666 //
667 // Save the Status because Image will get destroyed if it is unloaded.
668 //
669 Status = OcLoadedImage->Status;
670
671 //
672 // If the image returned an error, or if the image is an application
673 // unload it
674 //
675 if ( EFI_ERROR (OcLoadedImage->Status)
676 || (OcLoadedImage->Subsystem == EFI_IMAGE_SUBSYSTEM_EFI_APPLICATION))
677 {
678 InternalDirectUnloadImage (OcLoadedImage, ImageHandle);
679 }
680
681 return Status;
682}

692STATIC
693UINT32
694DetectCapabilities (
695 IN VOID *SourceBuffer,
696 IN UINT32 SourceSize
697 )
698{
699 BOOLEAN Exists;
700 UINT32 Result;
701
702 //
703 // Find Mac OS X version pattern.
704 // This pattern started to appear with 10.7.
705 //
706 Result = 0;
707 Exists = FindPattern (
708 (CONST UINT8 *)"Mac OS X 10.",
709 NULL,
710 L_STR_LEN ("Mac OS X 10."),
711 SourceBuffer,
712 SourceSize - sizeof (UINT32),
713 &Result
714 );
715
716 #ifdef MDE_CPU_IA32
717 //
718 // For IA32 mode the only question is whether we support K32_64.
719 // This starts with 10.7, and in theory is valid for some early
720 // developer preview 10.8 images, so simply decide on Mac OS X
721 // version pattern presence.
722 //
723 if (Exists) {
724 return OC_KERN_CAPABILITY_K32_U64;
725 }
726
727 return OC_KERN_CAPABILITY_K32_U32 | OC_KERN_CAPABILITY_K32_U64;
728 #else
729 //
730 // For X64 mode, when the pattern is found, this can be 10.7 or 10.8+.
731 // 10.7 supports K32_64 and K64, while newer versions have only K64.
732 //
733 if (Exists) {
734 if (((UINT8 *)SourceBuffer)[Result + L_STR_LEN ("Mac OS X 10.")] == '7') {
735 return OC_KERN_CAPABILITY_K32_U64 | OC_KERN_CAPABILITY_K64_U64;
736 }
737
738 return OC_KERN_CAPABILITY_K64_U64;
739 }
740
741 //
742 // The pattern is not found. This can be 10.6 or 10.4~10.5.
743 // 10.6 supports K32 and K64, while older versions have only K32.
744 // Detect 10.6 by x86_64 pattern presence.
745 //
746 Result = SourceSize / 2;
747 Exists = FindPattern (
748 (CONST UINT8 *)"x86_64",
749 NULL,
750 L_STR_SIZE ("x86_64"),
751 SourceBuffer,
752 SourceSize - sizeof (UINT32),
753 &Result
754 );
755 if (Exists) {
756 return OC_KERN_CAPABILITY_K32_U32 | OC_KERN_CAPABILITY_K32_U64 | OC_KERN_CAPABILITY_K64_U64;
757 }
758
759 return OC_KERN_CAPABILITY_K32_U32 | OC_KERN_CAPABILITY_K32_U64;
760 #endif
761}

763STATIC

765EFIAPI
766InternalEfiLoadImage (

774{

781 return EFI_INVALID_PARAMETER;
782 }
783
784 if ((SourceBuffer == NULL) && (DevicePath == NULL)) {

786 }
787

789 return EFI_UNSUPPORTED;

794 Status = InternalEfiLoadImageFile (
795 DevicePath,

798 );

801 DevicePath,

803 &SourceSize,
804 &SourceBuffer

807
808 if (!EFI_ERROR (Status)) {
809 AllocatedBuffer = SourceBuffer;
810 }
811 }
812
813 if ((DevicePath != NULL) && (SourceBuffer != NULL) && mImageLoaderEnabled) {
814 SecureBootStatus = OcAppleSecureBootVerify (
815 DevicePath,
816 SourceBuffer,
817 SourceSize
818 );
819 } else {
820 SecureBootStatus = EFI_UNSUPPORTED;
821 }
822
823 //
824 // A security violation means we should just die.
825 //
826 if (SecureBootStatus == EFI_SECURITY_VIOLATION) {
827 DEBUG ((
828 DEBUG_WARN,
829 "OCB: Apple Secure Boot prohibits this boot entry, enforcing!\n"
830 ));

834 //
835 // By default assume target default.
836 //
837 #ifdef MDE_CPU_IA32
838 mImageLoaderCaps = OC_KERN_CAPABILITY_K32_U32 | OC_KERN_CAPABILITY_K32_U64;
839 #else

843 if (SourceBuffer != NULL) {
844 RealSize = (UINT32)SourceSize;
845 #ifdef MDE_CPU_IA32
846 Status = FatFilterArchitecture32 ((UINT8 **)&SourceBuffer, &RealSize);
847 #else
848 Status = FatFilterArchitecture64 ((UINT8 **)&SourceBuffer, &RealSize);
849 #endif

851 //
852 // This is FAT image.
853 // Determine its capabilities.
854 //
855 if (!EFI_ERROR (Status) && (RealSize != SourceSize) && (RealSize >= EFI_PAGE_SIZE)) {
856 mImageLoaderCaps = DetectCapabilities (SourceBuffer, RealSize);
857 }

879 mImageLoaderPatch (
880 DevicePath,
881 SourceBuffer,
882 SourceSize
883 );
884 }

886 //
887 // Load the image ourselves in secure boot mode.
888 //

891 Status = OcImageLoaderLoad (
892 FALSE,
893 ParentImageHandle,
894 DevicePath,
895 SourceBuffer,
896 SourceSize,
897 ImageHandle
898 );
899 } else {

902 // This should not happen, just abort.
903 //
904 Status = EFI_UNSUPPORTED;
905 }
906 } else {
907 PreserveGrubShimHooks ();
908 Status = mOriginalEfiLoadImage (
909 BootPolicy,
910 ParentImageHandle,
911 DevicePath,
912 SourceBuffer,
913 SourceSize,
914 ImageHandle
915 );

917 }

925 // when loading via source buffer. Do it here.
926 //
927 if (!EFI_ERROR (Status) && (SourceBuffer != NULL) && (DevicePath != NULL)) {
928 InternalUpdateLoadedImage (*ImageHandle, DevicePath);
929 }
930
931 return Status;
