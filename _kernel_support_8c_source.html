45STATIC
46VOID
47ProtectRtMemoryFromRelocation (
48 IN OUT RT_RELOC_PROTECT_DATA *RtReloc,
49 IN UINTN MemoryMapSize,
50 IN UINTN DescriptorSize,
51 IN OUT EFI_MEMORY_DESCRIPTOR *MemoryMap,
52 IN EFI_PHYSICAL_ADDRESS SysTableArea,
53 IN UINTN SysTableAreaSize
54 )
55{
56 //
57 // We protect RT data & code from relocation by marking them MemMapIO except EFI_SYSTEM_TABLE area.
58 //
59 // This fixes NVRAM issues on some boards where access to NVRAM after boot services is possible
60 // only in SMM mode. RT driver passes data to SMM handler through previously negotiated buffer
61 // and this buffer must not be relocated.
62 // Explained and examined in detail by CodeRush and night199uk:
63 // https://web.archive.org/web/20141025080709/http://www.projectosx.com/forum/lofiversion/index.php/t3298.html
64 //
65 // Starting with APTIO V for NVRAM to work not only RT data but RT code too can no longer be moved
66 // due to the use of commbuffers. This, however, creates a memory protection issue, because
67 // XNU maps RT data as RW and code as RX, and AMI appears use global variables in some RT drivers.
68 // For this reason we shim (most?) affected RT services via wrapers that unset the WP bit during
69 // the UEFI call and set it back on return in a separate driver.
70 // Explained in detail by Download-Fritz and vit9696:
71 // http://www.insanelymac.com/forum/topic/331381-aptiomemoryfix (first 2 links in particular).
72 //
73 // EFI_SYSTEM_TABLE is passed directly through kernel boot arguments, and thus goes through static
74 // mapping (ml_static_ptovirt) in efi_set_tables_64 call. This mapping works as PHYS | CONST = VIRT.
75 // To avoid kernel accessing unmapped virtual address we let boot.efi relocate the page with
76 // EFI_SYSTEM_TABLE area. While technically it is possible to let the original page to be relocated,
77 // we pick a safer root by using a private copy.
78 //
79 // The primary downside of this approach is that boot.efi will still reserve the contiguous memory
80 // for runtime services after the kernel: efiRuntimeServicesPageCount pages starting from
81 // efiRuntimeServicesPageStart within kaddr ~ ksize range. However, unlike Macs, which have reserved
82 // gaps only for ACPI NVS, MemMapIO and similar regions, with this approach almost no physical memory
83 // in efiRuntimeServicesPageStart area is used at all. This memory is never reclaimed by XNU, which
84 // marks it as allocated in i386_vm_init. Expirements show that at least 85 MBs (Z170) are used for
85 // this process. On server systems the issue is much worse due to many devices in place.
86 // Ideally boot.efi should only count RT code and RT data pages, but it is not easy to change.
87 //
88
89 UINTN NumEntries;
90 UINTN Index;
91 EFI_MEMORY_DESCRIPTOR *Desc;
92 RT_RELOC_PROTECT_INFO *RelocInfo;
93
94 Desc = MemoryMap;
95 RtReloc->NumEntries = 0;
96 RelocInfo = &RtReloc->RelocInfo[0];
97 NumEntries = MemoryMapSize / DescriptorSize;
98
99 for (Index = 0; Index < NumEntries; ++Index) {
100 if ( ((Desc->Attribute & EFI_MEMORY_RUNTIME) != 0)
101 && (Desc->NumberOfPages > 0)
102 && ((Desc->Type == EfiRuntimeServicesCode) || (Desc->Type == EfiRuntimeServicesData))
103 && !AREA_WITHIN_DESCRIPTOR (Desc, SysTableArea, SysTableAreaSize))
104 {
105 if (RtReloc->NumEntries == ARRAY_SIZE (RtReloc->RelocInfo)) {
106 RUNTIME_DEBUG (
107 (
108 DEBUG_ERROR,
109 "OCABC: Cannot save mem type for entry: %Lx (type 0x%x)\n",
110 (UINT64)Desc->PhysicalStart,
111 (UINT32)Desc->Type
112 )
113 );
114 return;
115 }
116
117 RelocInfo->PhysicalStart = Desc->PhysicalStart;
118 RelocInfo->PhysicalEnd = LAST_DESCRIPTOR_ADDR (Desc);
119 RelocInfo->Type = Desc->Type;
120 Desc->Type = EfiMemoryMappedIO;
121 ++RelocInfo;
122 ++RtReloc->NumEntries;
123 }
124
125 Desc = NEXT_MEMORY_DESCRIPTOR (Desc, DescriptorSize);
126 }

141STATIC
142EFI_STATUS
143PerformRtMemoryVirtualMapping (
144 IN OUT KERNEL_SUPPORT_STATE *KernelState,
145 IN UINTN MemoryMapSize,
146 IN UINTN DescriptorSize,
147 IN UINT32 DescriptorVersion,
148 IN EFI_MEMORY_DESCRIPTOR *MemoryMap
149 )
150{
151 //
152 // About partial memmap:
153 // Some UEFIs are converting pointers to virtual addresses even if they do not
154 // point to regions with RT flag. This means that those UEFIs are using
155 // Desc->VirtualStart even for non-RT regions. Linux had issues with this:
156 // http://git.kernel.org/?p=linux/kernel/git/torvalds/linux-2.6.git;a=commit;h=7cb00b72876ea2451eb79d468da0e8fb9134aa8a
157 // They are doing it Windows way now - copying RT descriptors to separate
158 // mem map and passing that stripped map to SetVirtualAddressMap().
159 // We'll do the same, although it seems that just assigning
160 // VirtualStart = PhysicalStart for non-RT areas also does the job.
161 //
162 // About virtual to physical mappings:
163 // Also adds virtual to physical address mappings for RT areas. This is needed since
164 // SetVirtualAddressMap() does not work on my Aptio without that. Probably because some driver
165 // has a bug and is trying to access new virtual addresses during the change.
166 // Linux and Windows are doing the same thing and problem is
167 // not visible there.
168 //
169
170 UINTN NumEntries;
171 UINTN Index;
172 EFI_MEMORY_DESCRIPTOR *Desc;
173 EFI_MEMORY_DESCRIPTOR *VirtualDesc;
174 EFI_STATUS Status;
175 PAGE_MAP_AND_DIRECTORY_POINTER *PageTable;
176
177 Desc = MemoryMap;
178 NumEntries = MemoryMapSize / DescriptorSize;
179 VirtualDesc = KernelState->VmMap;
180 KernelState->VmMapSize = 0;
181 KernelState->VmMapDescSize = DescriptorSize;
182
183 //
184 // Get current VM page table.
185 //
186 PageTable = OcGetCurrentPageTable (NULL);
187
188 for (Index = 0; Index < NumEntries; ++Index) {
189 //
190 // Legacy note. Some UEFIs end up with "reserved" area with EFI_MEMORY_RUNTIME flag set when
191 // Intel HD3000 or HD4000 is used. For example, on GA-H81N-D2H there is a single 1 GB descriptor:
192 // 000000009F800000-00000000DF9FFFFF 0000000000040200 8000000000000000
193 //
194 // All known boot.efi starting from at least 10.5.8 properly handle this flag and do not assign
195 // virtual addresses to reserved descriptors. However, our legacy code had a bug, and did not
196 // check for EfiReservedMemoryType. Therefore it replaced such entries by EfiMemoryMappedIO
197 // to "prevent" boot.efi relocations.
198 //
199 // The relevant discussion and the original fix can be found here:
200 // http://web.archive.org/web/20141111124211/http://www.projectosx.com:80/forum/lofiversion/index.php/t2428-450.html
201 // https://sourceforge.net/p/cloverefiboot/code/605/
202 //
203 // The correct approach is to properly handle EfiReservedMemoryType with EFI_MEMORY_RUNTIME
204 // attribute set, and not mess with the memory map passed to boot.efi. As done here.
205 //
206 if ((Desc->Type != EfiReservedMemoryType) && ((Desc->Attribute & EFI_MEMORY_RUNTIME) != 0)) {
207 //
208 // Check if there is enough space in virtual map.
209 //
210 if (KernelState->VmMapSize + DescriptorSize > sizeof (KernelState->VmMap)) {
211 RUNTIME_DEBUG ((DEBUG_ERROR, "OCABC: Too many RT entries to memory map\n"));
212 return EFI_OUT_OF_RESOURCES;
213 }
214
215 //
216 // Copy region with EFI_MEMORY_RUNTIME flag to virtual map.
217 //
218 CopyMem (VirtualDesc, Desc, DescriptorSize);
219
220 //
221 // Define virtual to physical mapping.
222 //
223 Status = VmMapVirtualPages (
224 &KernelState->VmContext,
225 PageTable,
226 Desc->VirtualStart,
227 Desc->NumberOfPages,
228 Desc->PhysicalStart
229 );
230 if (EFI_ERROR (Status)) {
231 RUNTIME_DEBUG ((DEBUG_ERROR, "OCABC: RT mapping failure - %r\n", Status));
232 return EFI_OUT_OF_RESOURCES;
233 }
234
235 //
236 // Proceed to next virtual map slot.
237 //
238 VirtualDesc = NEXT_MEMORY_DESCRIPTOR (VirtualDesc, DescriptorSize);
239 KernelState->VmMapSize += DescriptorSize;
240 }
241
242 //
243 // Proceed to next original map slot.
244 //
245 Desc = NEXT_MEMORY_DESCRIPTOR (Desc, DescriptorSize);
246 }
247
248 VmFlushCaches ();
249
250 Status = gRT->SetVirtualAddressMap (
251 KernelState->VmMapSize,
252 DescriptorSize,
253 DescriptorVersion,
254 KernelState->VmMap
255 );
256
257 return Status;
258}

270RestoreProtectedRtMemoryTypes (
271 IN RT_RELOC_PROTECT_DATA *RtReloc,
272 IN UINTN MemoryMapSize,
273 IN UINTN DescriptorSize,
274 IN OUT EFI_MEMORY_DESCRIPTOR *MemoryMap
275 )
276{
277 UINTN Index;
278 UINTN Index2;
279 UINTN NumEntriesLeft;
280 UINTN NumEntries;
281 EFI_PHYSICAL_ADDRESS PhysicalStart;
282 EFI_PHYSICAL_ADDRESS PhysicalEnd;
283 EFI_MEMORY_DESCRIPTOR *Desc;
284
285 NumEntriesLeft = RtReloc->NumEntries;
286 NumEntries = MemoryMapSize / DescriptorSize;
287 Desc = MemoryMap;
288
289 for (Index = 0; Index < NumEntries && NumEntriesLeft > 0; ++Index) {
290 PhysicalStart = Desc->PhysicalStart;
291 PhysicalEnd = LAST_DESCRIPTOR_ADDR (Desc);
292
293 for (Index2 = 0; Index2 < RtReloc->NumEntries; ++Index2) {
294 //
295 // PhysicalStart match is enough, but just in case.
296 // Some types of firmware, such as on the Lenovo ThinkPad X240, have unusual reserved areas.
297 // For example, 0000000000000000-FFFFFFFFFFFFFFFF 0000000000000000 0000000000000000.
298 // Execute exact comparisons as fuzzy matching often results in errors.
299 //

301 && (PhysicalEnd == RtReloc->RelocInfo[Index2].PhysicalEnd))
302 {
303 Desc->Type = RtReloc->RelocInfo[Index2].Type;
304 --NumEntriesLeft;
305 break;
306 }
307 }
308
309 Desc = NEXT_MEMORY_DESCRIPTOR (Desc, DescriptorSize);
310 }
311
312 if (NumEntriesLeft > 0) {
313 RUNTIME_DEBUG (
314 (
315 DEBUG_ERROR,
316 "OCABC: Failed to restore %u entries out of %u\n",
317 (UINT32)NumEntriesLeft,
318 (UINT32)RtReloc->NumEntries
319 )
320 );
321 }

332AppleMapPrepareForBooting (
333 IN OUT BOOT_COMPAT_CONTEXT *BootCompat,
334 IN OUT VOID *BootArgs
335 )
336{
337 EFI_STATUS Status;
338 DTEntry Chosen;
339 CHAR8 *ArgsStr;
340 UINT32 ArgsSize;
341 OC_BOOT_ARGUMENTS BA;
342 UINTN MemoryMapSize;
343 EFI_MEMORY_DESCRIPTOR *MemoryMap;
344 UINTN DescriptorSize;
345
346 OcParseBootArgs (&BA, BootArgs);
347
348 if (BootCompat->Settings.ProvideCustomSlide) {
349 //
350 // Restore the variables we tampered with to support custom slides.
351 //
352 AppleSlideRestore (BootCompat, &BA);
353 }
354
355 if (BootCompat->Settings.DisableSingleUser) {
356 //
357 // First, there is a BootArgs entry for XNU.
358 //
359 OcRemoveArgumentFromCmd (BA.CommandLine, "-s");
360 }
361
362 if ( BootCompat->Settings.DisableSingleUser
363 || BootCompat->Settings.ForceBooterSignature)
364 {
365 DTInit ((VOID *)(UINTN)*BA.DeviceTreeP, BA.DeviceTreeLength);
366 Status = DTLookupEntry (NULL, "/chosen", &Chosen);
367 if (!EFI_ERROR (Status)) {
368 if (BootCompat->Settings.DisableSingleUser) {
369 //
370 // Second, there is a DT entry.
371 //
372 Status = DTGetProperty (Chosen, "boot-args", (VOID **)&ArgsStr, &ArgsSize);
373 if (!EFI_ERROR (Status) && (ArgsSize > 0)) {
374 OcRemoveArgumentFromCmd (ArgsStr, "-s");
375 }
376 }
377
378 if (BootCompat->Settings.ForceBooterSignature) {
379 Status = DTGetProperty (Chosen, "boot-signature", (VOID **)&ArgsStr, &ArgsSize);
380 if (!EFI_ERROR (Status) && (ArgsSize == SHA1_DIGEST_SIZE)) {
381 CopyMem (ArgsStr, BootCompat->Settings.BooterSignature, ArgsSize);
382 }
383 }
384 }
385 }
386
387 if (BootCompat->KernelState.RelocationBlock != 0) {
388 //
389 // When using Relocation Block EfiBoot will not virtualize the addresses since they
390 // cannot be mapped 1:1 due to any region from the relocation block being outside
391 // of static XNU vaddr to paddr mapping. This causes a clean early exit in their
392 // SetVirtualAddressMap calling routine avoiding gRT->SetVirtualAddressMap.
393 //
394 // For this reason we need to perform it ourselves right here before we restored
395 // runtime memory protections as we also need to defragment EFI_SYSTEM_TABLE memory
396 // to be accessible from XNU.
397 //
398 AppleRelocationVirtualize (BootCompat, &BA);
399 }
400
401 if (BootCompat->Settings.AvoidRuntimeDefrag) {
402 MemoryMapSize = *BA.MemoryMapSize;
403 MemoryMap = (EFI_MEMORY_DESCRIPTOR *)(UINTN)(*BA.MemoryMap);
404 DescriptorSize = *BA.MemoryMapDescriptorSize;
405
406 //
407 // We must restore EfiRuntimeServicesCode memory area types, because otherwise
408 // RuntimeServices won't be mapped.
409 //
410 RestoreProtectedRtMemoryTypes (
411 &BootCompat->RtReloc,
412 MemoryMapSize,
413 DescriptorSize,
414 MemoryMap
415 );
416
417 //
418 // On native Macs due to EfiBoot defragmentation it is guaranteed that
419 // VADDR % BASE_1GB == PADDR. macOS 11 started to rely on this in
420 // acpi_count_enabled_logical_processors, which needs to access MADT (APIC)
421 // ACPI table, and does that through ConfigurationTables.
422 //
423 // The simplest approach is to just copy the table, so that it is accessible
424 // at both actual mapping and 1:1 defragmented mapping. This should be safe,
425 // as the memory for 1:1 defragmented mapping is reserved by EfiBoot in the
426 // first place and is otherwise stolen anyway.
427 //
428 if (BootCompat->KernelState.ConfigurationTable != NULL) {
429 CopyMem (
430 (VOID *)((UINTN)BA.SystemTable->ConfigurationTable & (BASE_1GB - 1)),
431 BootCompat->KernelState.ConfigurationTable,
432 sizeof (*BootCompat->KernelState.ConfigurationTable) * BA.SystemTable->NumberOfTableEntries
433 );
434 }
435 }
436
437 if (BootCompat->KernelState.RelocationBlock != 0) {
438 AppleRelocationRebase (BootCompat, &BA);
439 }

449VOID
450AppleMapPrepareForHibernateWake (
451 IN OUT BOOT_COMPAT_CONTEXT *BootCompat,
452 IN UINTN ImageHeaderPage
453 )
454{
455 IOHibernateImageHeader *ImageHeader;
456 IOHibernateHandoff *Handoff;
457
458 ImageHeader = (IOHibernateImageHeader *)EFI_PAGES_TO_SIZE (ImageHeaderPage);
459
460 //
461 // Legacy note. In legacy implementations systemTableOffset was unconditionally overwritten
462 // with a wrong address due to ImageHeader->runtimePages not being converted from pages to bytes.
463 // Fortunately systemTableOffset was unused when kIOHibernateHandoffTypeMemoryMap is unspecified.
464 // systemTableOffset is calculated properly by boot.efi itself starting from 10.6.8 at least,
465 // and thus this assignment was useless in the first place.
466 //
467
468 //
469 // At this step we have two routes.
470 //
471 // 1. Remove newly generated memory map from hibernate image to let XNU use the original mapping.
472 // This is known to work well on most systems primarily because Windows requires UEFI firmware
473 // to preserve physical memory consistency at S4 wake. "On a UEFI platform, firmware runtime memory
474 // must be consistent across S4 sleep state transitions, in both size and location.", see:
475 // https://docs.microsoft.com/en-us/windows-hardware/design/device-experiences/oem-uefi#hibernation-state-s4-transition-requirements
476 // 2. Recover memory map just as we do for normal booting. This created issues on some types of firmware
477 // resulting in unusual memory maps after S4 wake. In other cases, this should not immediately
478 // break things. XNU will entirely remove efiRuntimeServicesPageStart/efiRuntimeServicesPageSize
479 // mapping and our new memory map entries will unconditionally overwrite previous ones. In case
480 // no physical memory changes happened, this should work fine.
481 //
482 Handoff = (IOHibernateHandoff *)EFI_PAGES_TO_SIZE ((UINTN)ImageHeader->handoffPages);
483 while (Handoff->type != kIOHibernateHandoffTypeEnd) {
484 if (Handoff->type == kIOHibernateHandoffTypeMemoryMap) {
485 if (BootCompat->Settings.DiscardHibernateMap) {
486 //
487 // Route 1. Discard the new memory map here, and let XNU use what it had.
488 // It is unknown whether there are still examples of firmware that need this.
489 //
490 Handoff->type = kIOHibernateHandoffType;
491 } else {
492 //
493 // Route 2. Recovery memory protection types just as normal boot.
494 //
495
496 if (BootCompat->KernelState.VmMapDescSize == 0) {
497 RUNTIME_DEBUG ((DEBUG_ERROR, "OCABC: Saved descriptor size cannot be 0\n"));
498 return;
499 }
500
501 //
502 // TODO: If we try to work on hibernation support with relocation block
503 // We will need to add a call similar to AppleRelocationVirtualize here.
504 //
505
506 if (BootCompat->Settings.AvoidRuntimeDefrag) {
507 //
508 // I think we should not be there, but ideally all quirks are relatively independent.
509 //
510 RestoreProtectedRtMemoryTypes (
511 &BootCompat->RtReloc,
512 Handoff->bytecount,
513 BootCompat->KernelState.VmMapDescSize,
514 (EFI_MEMORY_DESCRIPTOR *)(UINTN)Handoff->data
515 );
516 }
517 }
518
519 break;
520 }
521
522 Handoff = (IOHibernateHandoff *)((UINTN)Handoff + sizeof (Handoff) + Handoff->bytecount);
523 }
524
525 //
526 // TODO: To support hibernation with relocation block we will need to add a call similar
527 // to AppleRelocationRebase here.
528 //
529}

621VOID
622AppleMapPrepareKernelJump (
623 IN OUT BOOT_COMPAT_CONTEXT *BootCompat,
624 IN EFI_PHYSICAL_ADDRESS CallGate,
625 IN UINTN HookAddress
626 )
627{
628 CALL_GATE_JUMP *CallGateJump;
629
630 //
631 // There is no reason to patch the kernel when we do not need it.
632 //
633 if ( !BootCompat->Settings.AvoidRuntimeDefrag
634 && !BootCompat->Settings.DiscardHibernateMap
635 && !BootCompat->Settings.AllowRelocationBlock
636 && !BootCompat->Settings.DisableSingleUser
637 && !BootCompat->Settings.ForceBooterSignature)
638 {
639 return;
640 }
641
642 #ifndef MDE_CPU_X64
643 RUNTIME_DEBUG ((DEBUG_ERROR, "OCABC: Kernel trampolines are unsupported for non-X64\n"));
644 CpuDeadLoop ();
645 #endif
646
647 //
648 // Check whether we have address and abort if not.
649 //
650 if (CallGate == 0) {
651 RUNTIME_DEBUG ((DEBUG_ERROR, "OCABC: Failed to find call gate address\n"));
652 return;
653 }
654
655 CallGateJump = (VOID *)(UINTN)CallGate;
656
657 //
658 // Move call gate jump bytes front.
659 // Performing this on the EfiBootRt KCG may bork the binary, but right now
660 // only corrupts an unused string.
661 //
662 CopyMem (
663 CallGateJump + 1,
664 CallGateJump,
665 ESTIMATED_CALL_GATE_SIZE
666 );
667 //
668 // lea r8, [rip+XXX]
669 // Passes KCG as third argument to be relocatable. macOS 13 Developer Beta 1
670 // copies the KCG into a separately allocated buffer.
671 //
672 CallGateJump->LeaRip.Command[0] = 0x4C;
673 CallGateJump->LeaRip.Command[1] = 0x8D;
674 CallGateJump->LeaRip.Command[2] = 0x05;
675 CallGateJump->LeaRip.Argument = sizeof (*CallGateJump) - sizeof (CallGateJump->LeaRip);
676 CallGateJump->Jmp.Command = 0x25FF;
677 CallGateJump->Jmp.Argument = 0x0;
678 CallGateJump->Jmp.Address = HookAddress;
679}
