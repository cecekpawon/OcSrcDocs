89

93 Status = EFI_NOT_FOUND;
94 }
95
96 if (Fallback && EFI_ERROR (Status)) {
97 if (AsciiStrCmp (BaseType, OC_VOICE_OVER_AUDIO_BASE_TYPE_APPLE) == 0) {
98 if (AsciiStrCmp (BasePath, APPLE_VOICE_OVER_AUDIO_FILE_BEEP) == 0) {
99 Status = OcPlayAudioBeep (
100 Context,
101 OC_VOICE_OVER_SIGNALS_NORMAL,
102 OC_VOICE_OVER_SIGNAL_NORMAL_MS,
103 OC_VOICE_OVER_SILENCE_NORMAL_MS
104 );
105 }
106 } else if (AsciiStrCmp (BaseType, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE) == 0) {
107 if (AsciiStrCmp (BasePath, OC_VOICE_OVER_AUDIO_FILE_ENTER_PASSWORD) == 0) {
108 Status = OcPlayAudioBeep (
109 Context,
110 OC_VOICE_OVER_SIGNALS_PASSWORD,
111 OC_VOICE_OVER_SIGNAL_NORMAL_MS,
112 OC_VOICE_OVER_SILENCE_NORMAL_MS
113 );
114 } else if (AsciiStrCmp (BasePath, OC_VOICE_OVER_AUDIO_FILE_PASSWORD_ACCEPTED) == 0) {
115 Status = OcPlayAudioBeep (
116 Context,
117 OC_VOICE_OVER_SIGNALS_PASSWORD_OK,
118 OC_VOICE_OVER_SIGNAL_NORMAL_MS,
119 OC_VOICE_OVER_SILENCE_NORMAL_MS
120 );
121 } else if (AsciiStrCmp (BasePath, OC_VOICE_OVER_AUDIO_FILE_PASSWORD_INCORRECT) == 0) {
122 Status = OcPlayAudioBeep (
123 Context,
124 OC_VOICE_OVER_SIGNALS_ERROR,
125 OC_VOICE_OVER_SIGNAL_ERROR_MS,
126 OC_VOICE_OVER_SILENCE_ERROR_MS
127 );

129 Status = OcPlayAudioBeep (

131 OC_VOICE_OVER_SIGNALS_HWERROR,
132 OC_VOICE_OVER_SIGNAL_ERROR_MS,
133 OC_VOICE_OVER_SILENCE_ERROR_MS
134 );

136 Status = OcPlayAudioBeep (

138 OC_VOICE_OVER_SIGNALS_ERROR,
139 OC_VOICE_OVER_SIGNAL_ERROR_MS,
140 OC_VOICE_OVER_SIGNAL_NORMAL_MS
