156//
157// Specify PAT type at index in PAT MSR.
158//
159#define SET_PAT_N(PatIndex, PatType) (LShiftU64 ((PatType) * 1ULL, (PatIndex) * 8))

161//
162// Get PAT type at index in PAT MSR.

168//
