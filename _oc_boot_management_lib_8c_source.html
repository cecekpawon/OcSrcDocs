46RunShowMenu (
47 IN OC_BOOT_CONTEXT *BootContext,
48 OUT OC_BOOT_ENTRY **ChosenBootEntry
49 )
50{
51 EFI_STATUS Status;
52 OC_BOOT_ENTRY **BootEntries;
53 UINT32 EntryReason;
54
55 if ( !BootContext->PickerContext->ApplePickerUnsupported
56 && (BootContext->PickerContext->PickerMode == OcPickerModeApple))
57 {
58 Status = OcRunFirmwareApplication (&gAppleBootPickerFileGuid, TRUE);
59 //
60 // This should not return on success.
61 //
62 DEBUG ((DEBUG_INFO, "OCB: Apple BootPicker failed on error - %r, fallback to builtin\n", Status));
63 BootContext->PickerContext->ApplePickerUnsupported = TRUE;
64 }
65
66 BootEntries = OcEnumerateEntries (BootContext);
67 if (BootEntries == NULL) {
68 return EFI_OUT_OF_RESOURCES;
69 }
70
71 //
72 // We are not allowed to have no default entry.
73 // However, if default entry is a tool or a system entry, never autoboot it.
74 //
75 if (BootContext->DefaultEntry == NULL) {
76 BootContext->DefaultEntry = BootEntries[0];
77 BootContext->PickerContext->TimeoutSeconds = 0;
78 }
79
80 //
81 // Ensure that picker entry reason is set as it can be read by boot.efi.
82 //
83 EntryReason = ApplePickerEntryReasonUnknown;
84 gRT->SetVariable (
85 APPLE_PICKER_ENTRY_REASON_VARIABLE_NAME,
86 &gAppleVendorVariableGuid,
87 EFI_VARIABLE_BOOTSERVICE_ACCESS,
88 sizeof (EntryReason),
89 &EntryReason
90 );
91
92 Status = OcInitHotKeys (BootContext->PickerContext);
93 if (EFI_ERROR (Status)) {
94 FreePool (BootEntries);
95 return Status;
96 }
97
98 Status = BootContext->PickerContext->ShowMenu (
99 BootContext,
100 BootEntries,
101 ChosenBootEntry
102 );
103
104 OcFreeHotKeys (BootContext->PickerContext);
105
106 FreePool (BootEntries);
107
108 return Status;
109}

172OcRunBootPicker (
173 IN OC_PICKER_CONTEXT *Context
174 )
175{
176 EFI_STATUS Status;
177 APPLE_KEY_MAP_AGGREGATOR_PROTOCOL *KeyMap;
178 OC_BOOT_CONTEXT *BootContext;
179 OC_BOOT_ENTRY *Chosen;
180 BOOLEAN SaidWelcome;
181
182 SaidWelcome = FALSE;
183
184 OcImageLoaderActivate ();
185
186 KeyMap = OcAppleKeyMapInstallProtocols (FALSE);
187 if (KeyMap == NULL) {
188 DEBUG ((DEBUG_ERROR, "OCB: AppleKeyMap locate failure\n"));
189 return EFI_NOT_FOUND;
190 }
191
192 //
193 // This one is handled as is for Apple BootPicker for now.
194 //
195 if (Context->PickerCommand != OcPickerDefault) {
196 Status = InternalRunRequestPrivilege (Context, OcPrivilegeAuthorized);
197 if (EFI_ERROR (Status)) {
198 if (Status != EFI_ABORTED) {
199 ASSERT (FALSE);
200 return Status;
201 }
202
203 Context->PickerCommand = OcPickerDefault;
204 }
205 }
206
207 if ((Context->PickerCommand == OcPickerShowPicker) && (Context->PickerMode == OcPickerModeApple)) {
208 Status = OcRunFirmwareApplication (&gAppleBootPickerFileGuid, TRUE);
209 DEBUG ((DEBUG_INFO, "OCB: Apple BootPicker failed - %r, fallback to builtin\n", Status));
210 Context->ApplePickerUnsupported = TRUE;
211 }
212
213 if ((Context->PickerCommand != OcPickerShowPicker) && (Context->PickerCommand != OcPickerDefault)) {
214 //
215 // We cannot ignore auxiliary entries for all other modes.
216 //
217 Context->HideAuxiliary = FALSE;
218 }
219
220 while (TRUE) {
221 //
222 // Turbo-boost scanning when bypassing picker.
223 //
224 if ((Context->PickerCommand == OcPickerDefault) || (Context->PickerCommand == OcPickerProtocolHotKey)) {
225 BootContext = OcScanForDefaultBootEntry (Context);
226 } else {
227 ASSERT (
228 Context->PickerCommand == OcPickerShowPicker
229 || Context->PickerCommand == OcPickerBootApple
230 || Context->PickerCommand == OcPickerBootAppleRecovery
231 );
232
233 BootContext = OcScanForBootEntries (Context);
234 }
235
236 //
237 // We have no entries at all or have auxiliary entries.
238 // Fallback to showing menu in the latter case.
239 //
240 if (BootContext == NULL) {
241 //
242 // TODO: Failed protocol hotkey can access OcPickerShowPicker mode even if
243 // this is denied by InternalRunRequestPrivilege above, is this OK?
244 //
245 if (Context->HideAuxiliary || (Context->PickerCommand == OcPickerProtocolHotKey)) {
246 DEBUG ((DEBUG_INFO, "OCB: System has no boot entries, showing picker with auxiliary\n"));
247 Context->PickerCommand = OcPickerShowPicker;
248 Context->HideAuxiliary = FALSE;
249 continue;
250 }
251
252 DEBUG ((DEBUG_WARN, "OCB: System has no boot entries\n"));
253 return EFI_NOT_FOUND;
254 }
255
256 if (Context->PickerCommand == OcPickerShowPicker) {
257 DEBUG ((
258 DEBUG_INFO,
259 "OCB: Showing menu... %a\n",
260 Context->PollAppleHotKeys ? "(polling hotkeys)" : ""
261 ));
262
263 if (!SaidWelcome) {
264 OcPlayAudioFile (Context, OC_VOICE_OVER_AUDIO_FILE_WELCOME, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE, FALSE);
265 SaidWelcome = TRUE;
266 }
267
268 Status = RunShowMenu (BootContext, &Chosen);
269
270 if (EFI_ERROR (Status) && (Status != EFI_ABORTED)) {
271 if (BootContext->PickerContext->ShowMenu != OcShowSimpleBootMenu) {
272 DEBUG ((DEBUG_WARN, "OCB: External interface ShowMenu failure, fallback to builtin - %r\n", Status));
273 BootContext->PickerContext->ShowMenu = OcShowSimpleBootMenu;
274 BootContext->PickerContext->RequestPrivilege = OcShowSimplePasswordRequest;
275 Status = RunShowMenu (BootContext, &Chosen);
276 }
277 }
278
279 if (EFI_ERROR (Status) && (Status != EFI_ABORTED)) {
280 DEBUG ((DEBUG_ERROR, "OCB: ShowMenu failed - %r\n", Status));
281 OcFreeBootContext (BootContext);
282 return Status;
283 }
284 } else if (BootContext->DefaultEntry != NULL) {
285 Chosen = BootContext->DefaultEntry;
286 Status = EFI_SUCCESS;
287 } else {
288 //
289 // This can only be failed macOS or macOS recovery boot.
290 // We may actually not rescan here.
291 //
292 ASSERT (Context->PickerCommand == OcPickerBootApple || Context->PickerCommand == OcPickerBootAppleRecovery);
293 DEBUG ((DEBUG_INFO, "OCB: System has no default boot entry, showing menu\n"));
294 Context->PickerCommand = OcPickerShowPicker;
295 OcFreeBootContext (BootContext);
296 continue;
297 }
298
299 ASSERT (!EFI_ERROR (Status) || Status == EFI_ABORTED);
300
301 Context->TimeoutSeconds = 0;
302
303 if (!EFI_ERROR (Status)) {
304 DEBUG ((
305 DEBUG_INFO,
306 "OCB: Should boot from %u. %s (T:%d|F:%d|G:%d|E:%d|DEF:%d)\n",
307 Chosen->EntryIndex,
308 Chosen->Name,
309 Chosen->Type,
310 Chosen->IsFolder,
311 Chosen->IsGeneric,
312 Chosen->IsExternal,
313 Chosen->SetDefault
314 ));
315
316 if (Context->PickerCommand == OcPickerShowPicker) {
317 ASSERT (Chosen->EntryIndex > 0);
318
319 if (Chosen->SetDefault) {
320 if (Context->PickerCommand == OcPickerShowPicker) {
321 OcPlayAudioFile (Context, OC_VOICE_OVER_AUDIO_FILE_SELECTED, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE, FALSE);
322 OcPlayAudioFile (Context, OC_VOICE_OVER_AUDIO_FILE_DEFAULT, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE, FALSE);
323 OcPlayAudioEntry (Context, Chosen);
324 }
325
326 Status = OcSetDefaultBootEntry (Context, Chosen);
327 DEBUG ((DEBUG_INFO, "OCB: Setting default - %r\n", Status));
328 }
329
330 //
331 // Clear screen of previous console contents - e.g. from builtin picker,
332 // log messages or previous console tool - before loading the entry.
333 //
334 gST->ConOut->ClearScreen (gST->ConOut);
335 if (Context->ShowMenu == OcShowSimpleBootMenu) {
336 gST->ConOut->TestString (gST->ConOut, OC_CONSOLE_MARK_UNCONTROLLED);
337 }
338
339 //
340 // Voice chosen information.
341 //
342 OcPlayAudioFile (Context, OC_VOICE_OVER_AUDIO_FILE_LOADING, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE, FALSE);
343 Status = OcPlayAudioEntry (Context, Chosen);
344 if (EFI_ERROR (Status)) {
345 OcPlayAudioBeep (
346 Context,
347 OC_VOICE_OVER_SIGNALS_PASSWORD_OK,
348 OC_VOICE_OVER_SIGNAL_NORMAL_MS,
349 OC_VOICE_OVER_SILENCE_NORMAL_MS
350 );
351 }
352 }
353
354 Status = OcLoadBootEntry (
355 Context,
356 Chosen,
357 gImageHandle
358 );
359
360 //
361 // Do not wait on successful return code.
362 //
363 if (EFI_ERROR (Status)) {
364 OcPlayAudioFile (Context, OC_VOICE_OVER_AUDIO_FILE_EXECUTION_FAILURE, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE, TRUE);
365 gBS->Stall (SECONDS_TO_MICROSECONDS (3));
366 //
367 // Show picker on first failure.
368 //
369 Context->PickerCommand = OcPickerShowPicker;
370 } else {
371 OcPlayAudioFile (Context, OC_VOICE_OVER_AUDIO_FILE_EXECUTION_SUCCESSFUL, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE, FALSE);
372 }
373
374 //
375 // Ensure that we flush all pressed keys after the application.
376 // This resolves the problem of application-pressed keys being used to control the menu.
377 //
378 OcKeyMapFlush (KeyMap, 0, TRUE);
379 }
380
381 OcFreeBootContext (BootContext);
382 }
383}
