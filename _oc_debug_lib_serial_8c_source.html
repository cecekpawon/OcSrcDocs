108 CHAR8 Buffer[MAX_DEBUG_MESSAGE_LENGTH];
109
110 //
111 // If Format is NULL, then ASSERT().
112 //
113 ASSERT (Format != NULL);
114
115 //
116 // Check driver debug mask value and global mask
117 //

120 }
