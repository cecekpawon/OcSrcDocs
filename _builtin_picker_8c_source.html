179 } else {
180 Code[0] = L' ';
181 }

185 //

206 }

316 OcPlayAudioFile (BootContext->PickerContext, OC_VOICE_OVER_AUDIO_FILE_RESTART, OC_VOICE_OVER_AUDIO_BASE_TYPE_OPEN_CORE, FALSE);
317 }
318 }
319 }

321 #if defined (BUILTIN_DEMONSTRATE_TYPING)

323 gST->ConOut->SetCursorPosition (gST->ConOut, TypingColumn, TypingRow);
324 Code[0] = IsEntering ? L'_' : ' ';

327 #endif
328}
