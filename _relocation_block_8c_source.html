40AppleRelocationAllocatePages (
41 IN OUT BOOT_COMPAT_CONTEXT *BootCompat,
42 IN EFI_GET_MEMORY_MAP GetMemoryMap,
43 IN EFI_ALLOCATE_PAGES AllocatePages,
44 IN UINTN NumberOfPages,
45 IN OUT EFI_PHYSICAL_ADDRESS *Memory
46 )
47{
48 EFI_STATUS Status;
49 UINTN EssentialSize;
50
51 EssentialSize = AppleSlideGetRelocationSize (BootCompat);
52 if (EssentialSize == 0) {
53 return EFI_UNSUPPORTED;
54 }
55
56 //
57 // Operating systems up to macOS 10.15 allocate starting with TEXT segment (2MB).
58 // macOS 11.0 allocates starting with HIB segment (1MB), but it does not support
59 // this quirk anyway due to AllocatePages in AllocateAddress mode Address pointer
60 // no longer being reread after the allocation in EfiBoot.
61 //
62 if ((*Memory == KERNEL_TEXT_PADDR) && (BootCompat->KernelState.RelocationBlock == 0)) {
63 BootCompat->KernelState.RelocationBlock = BASE_4GB;
64 Status = OcAllocatePagesFromTop (
65 EfiLoaderData,
66 EFI_SIZE_TO_PAGES (EssentialSize),
67 &BootCompat->KernelState.RelocationBlock,
68 GetMemoryMap,
69 AllocatePages,
70 NULL
71 );
72 if (EFI_ERROR (Status)) {
73 DEBUG ((
74 DEBUG_INFO,
75 "OCABC: Relocation block (0x%Lx) allocation failure - %r\n",
76 EssentialSize,
77 Status
78 ));
79 return Status;
80 }
81
82 BootCompat->KernelState.RelocationBlockUsed = 0;
83 }
84
85 //
86 // Not our allocation.
87 //
88 if ( (BootCompat->KernelState.RelocationBlock == 0)
89 || (*Memory < KERNEL_BASE_PADDR)
90 || (*Memory >= KERNEL_BASE_PADDR + EssentialSize))
91 {
92 return EFI_INVALID_PARAMETER;
93 }
94
95 //
96 // Track actually occupied memory.
97 //
98 EssentialSize = (UINTN)(*Memory - KERNEL_BASE_PADDR + EFI_PAGES_TO_SIZE (NumberOfPages));
99 if (EssentialSize > BootCompat->KernelState.RelocationBlockUsed) {
100 BootCompat->KernelState.RelocationBlockUsed = EssentialSize;
101 }
102
103 //
104 // Assume that EfiBoot does not try to reallocate memory.
105 //
106 *Memory = *Memory - KERNEL_BASE_PADDR + BootCompat->KernelState.RelocationBlock;
107
108 return EFI_SUCCESS;
109}

140AppleRelocationVirtualize (
141 IN OUT BOOT_COMPAT_CONTEXT *BootCompat,
142 IN OUT OC_BOOT_ARGUMENTS *BA
143 )
144{
145 EFI_STATUS Status;
146 UINTN MemoryMapSize;
147 UINTN DescriptorSize;
148 UINT32 DescriptorVersion;
149 EFI_MEMORY_DESCRIPTOR *MemoryMap;
150 EFI_PHYSICAL_ADDRESS KernelRTAddress;
151 UINTN NumEntries;
152 UINTN Index;
153 EFI_MEMORY_DESCRIPTOR *Desc;
154 UINTN BlockSize;
155 UINT8 *KernelRTBlock;
156
157 MemoryMapSize = *BA->MemoryMapSize;
158 DescriptorSize = *BA->MemoryMapDescriptorSize;
159 DescriptorVersion = *BA->MemoryMapDescriptorVersion;
160 MemoryMap = (EFI_MEMORY_DESCRIPTOR *)(UINTN)*BA->MemoryMap;
161 KernelRTAddress = EFI_PAGES_TO_SIZE (*BA->RuntimeServicesPG)
162 - (UINT32)(BootCompat->KernelState.RelocationBlock - KERNEL_BASE_PADDR);
163
164 //
165 // (1) Assign virtual addresses to all runtime blocks (but reserved).
166 //
167 Desc = MemoryMap;
168 NumEntries = MemoryMapSize / DescriptorSize;
169
170 for (Index = 0; Index < NumEntries; ++Index) {
171 BlockSize = EFI_PAGES_TO_SIZE ((UINTN)Desc->NumberOfPages);
172
173 if (Desc->Type == EfiReservedMemoryType) {
174 Desc->Attribute &= ~EFI_MEMORY_RUNTIME;
175 } else if ((Desc->Attribute & EFI_MEMORY_RUNTIME) != 0) {
176 Desc->VirtualStart = KernelRTAddress + KERNEL_STATIC_VADDR;
177 KernelRTAddress += BlockSize;
178 }
179
180 Desc = NEXT_MEMORY_DESCRIPTOR (Desc, DescriptorSize);
181 }
182
183 //
184 // (2) Transition to virtual memory.
185 //
186 Status = gRT->SetVirtualAddressMap (
187 MemoryMapSize,
188 DescriptorSize,
189 DescriptorVersion,
190 MemoryMap
191 );
192
193 //
194 // (3) Perform quick dirty defragmentation similarly to EfiBoot to make vaddr = paddr
195 // for critical areas like EFI_SYSTEM_TABLE.
196 //
197 Desc = MemoryMap;
198
199 for (Index = 0; Index < NumEntries; ++Index) {
200 if ((Desc->Type == EfiRuntimeServicesCode) || (Desc->Type == EfiRuntimeServicesData)) {
201 //
202 // Get physical address from statically mapped virtual.
203 //
204 KernelRTBlock = (UINT8 *)(UINTN)(Desc->VirtualStart & (BASE_1GB - 1));
205 BlockSize = EFI_PAGES_TO_SIZE ((UINTN)Desc->NumberOfPages);
206 CopyMem (
207 KernelRTBlock + (BootCompat->KernelState.RelocationBlock - KERNEL_BASE_PADDR),
208 (VOID *)(UINTN)Desc->PhysicalStart,
209 BlockSize
210 );
211
212 ZeroMem ((VOID *)(UINTN)Desc->PhysicalStart, BlockSize);
213
214 //
215 // (4) Sync changes to EFI_SYSTEM_TABLE location with boot args.
216 //
217 if ( (Desc->PhysicalStart <= *BA->SystemTableP)
218 && (*BA->SystemTableP <= LAST_DESCRIPTOR_ADDR (Desc)))
219 {
220 *BA->SystemTableP = (UINT32)((UINTN)KernelRTBlock
221 + (*BA->SystemTableP - Desc->PhysicalStart)
222 + (BootCompat->KernelState.RelocationBlock - KERNEL_BASE_PADDR));
223 }
224
225 //
226 // Mark old RT block in MemMap as free memory and remove RT attribute.
227 //
228 Desc->Type = EfiConventionalMemory;
229 Desc->Attribute = Desc->Attribute & (~EFI_MEMORY_RUNTIME);
230 }
231
232 Desc = NEXT_MEMORY_DESCRIPTOR (Desc, DescriptorSize);
233 }
234
235 return Status;
236}

239AppleRelocationRebase (

246 CHAR8 *PropName;
247 DTMemMapEntry *PropValue;
248 OpaqueDTPropertyIterator OPropIter;
249 DTPropertyIterator PropIter;
250 UINT32 RelocDiff;
251
252 PropIter = &OPropIter;
253
254 RelocDiff = (UINT32)(BootCompat->KernelState.RelocationBlock - KERNEL_BASE_PADDR);
255
256 DTInit ((DTEntry)(UINTN)*BA->DeviceTreeP, BA->DeviceTreeLength);
257 Status = DTLookupEntry (NULL, "/chosen/memory-map", &MemMap);
258
259 if (!EFI_ERROR (Status)) {
260 Status = DTCreatePropertyIterator (MemMap, &OPropIter);
261 if (!EFI_ERROR (Status)) {
262 while (!EFI_ERROR (DTIterateProperties (PropIter, &PropName))) {
263 //
264 // /chosen/memory-map props have DTMemMapEntry values (address, length).
265 // We need to correct the addresses in matching types.
266 //
267
268 //
269 // Filter entries with different size right away.
270 //
271 if (PropIter->CurrentProperty->Length != sizeof (DTMemMapEntry)) {
272 continue;
273 }
274
275 //
276 // Filter enteries out of the relocation range.
277 //
278 PropValue = (DTMemMapEntry *)((UINT8 *)PropIter->CurrentProperty + sizeof (DTProperty));
279 if ( (PropValue->Address < BootCompat->KernelState.RelocationBlock)
280 || (PropValue->Address >= BootCompat->KernelState.RelocationBlock + BootCompat->KernelState.RelocationBlockUsed))
281 {
282 continue;
283 }
284
285 //
286 // Patch the addresses up.
287 //
288 PropValue->Address -= RelocDiff;
289 }
290 }
291 }
292
293 *BA->MemoryMap -= RelocDiff;
294 *BA->KernelAddrP -= RelocDiff;
295 *BA->SystemTableP -= RelocDiff;
296 *BA->RuntimeServicesPG -= EFI_SIZE_TO_PAGES (RelocDiff);
297 //
298 // Note, this one does not seem to be used by XNU but we set it anyway.
299 //
300 *BA->RuntimeServicesV = EFI_PAGES_TO_SIZE (*BA->RuntimeServicesPG) + KERNEL_STATIC_VADDR;
301 *BA->DeviceTreeP -= RelocDiff;
302}
