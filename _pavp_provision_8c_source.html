86 &gEfiFirmwareVolumeProtocolGuid,
87 (VOID **)&FirmwareVolumeInterface
88 );
89

100 FvNameGuid,
101 EFI_SECTION_RAW,

103 Buffer,
104 BufferSize,
105 &AuthenticationStatus
106 );

108 if (!EFI_ERROR (Status)) {
109 gBS->FreePool (HandleBuffer);
110 return Status;
111 }
112 }
113
114 gBS->FreePool (HandleBuffer);

116 }

118 //
119 // Implement fallback for our firmware.
120 //
121 if (EFI_ERROR (Status)) {
122 DEBUG ((DEBUG_INFO, "OCPAVP: No %g in firmware, using default - %r\n", FvNameGuid, Status));
123

125 *Buffer = AllocateCopyPool (gDefaultAppleEpidCertificateSize, gDefaultAppleEpidCertificate);
126 *BufferSize = gDefaultAppleEpidCertificateSize;
127 } else if (CompareGuid (&gAppleEpidGroupPublicKeysFileGuid, FvNameGuid)) {
128 *Buffer = AllocateCopyPool (gDefaultAppleGroupPublicKeysSize, gDefaultAppleGroupPublicKeys);
129 *BufferSize = gDefaultAppleGroupPublicKeysSize;
130 } else {
131 *Buffer = NULL;
132 }
133
134 if (*Buffer != NULL) {
135 Status = EFI_SUCCESS;
136 } else {
137 Status = EFI_NOT_FOUND;
138 }
139 }
140
