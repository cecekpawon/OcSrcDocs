73OcGetPhysicalAddress (
74 IN PAGE_MAP_AND_DIRECTORY_POINTER *PageTable OPTIONAL,
75 IN EFI_VIRTUAL_ADDRESS VirtualAddr,
76 OUT EFI_PHYSICAL_ADDRESS *PhysicalAddr
77 )
78{
79 EFI_PHYSICAL_ADDRESS Start;
80 VIRTUAL_ADDR VA;
81 VIRTUAL_ADDR VAStart;
82 VIRTUAL_ADDR VAEnd;
83 PAGE_MAP_AND_DIRECTORY_POINTER *PML4;
84 PAGE_MAP_AND_DIRECTORY_POINTER *PDPE;
85 PAGE_MAP_AND_DIRECTORY_POINTER *PDE;
86 PAGE_TABLE_4K_ENTRY *PTE4K;
87 PAGE_TABLE_2M_ENTRY *PTE2M;
88 PAGE_TABLE_1G_ENTRY *PTE1G;
89
90 if (PageTable == NULL) {
91 PageTable = OcGetCurrentPageTable (NULL);
92 }
93
94 VA.Uint64 = (UINT64)VirtualAddr;
95
96 //
97 // PML4
98 //
99 PML4 = PageTable;
100 PML4 += VA.Pg4K.PML4Offset;
101 VAStart.Uint64 = 0;
102 VAStart.Pg4K.PML4Offset = VA.Pg4K.PML4Offset;
103 VA_FIX_SIGN_EXTEND (VAStart);
104 VAEnd.Uint64 = ~(UINT64)0;
105 VAEnd.Pg4K.PML4Offset = VA.Pg4K.PML4Offset;
106 VA_FIX_SIGN_EXTEND (VAEnd);
107
108 if (!PML4->Bits.Present) {
109 return EFI_NO_MAPPING;
110 }
111
112 //
113 // PDPE
114 //
115 PDPE = (PAGE_MAP_AND_DIRECTORY_POINTER *)(UINTN)(PML4->Uint64 & PAGING_4K_ADDRESS_MASK_64);
116 PDPE += VA.Pg4K.PDPOffset;
117 VAStart.Pg4K.PDPOffset = VA.Pg4K.PDPOffset;
118 VAEnd.Pg4K.PDPOffset = VA.Pg4K.PDPOffset;
119
120 if (!PDPE->Bits.Present) {
121 return EFI_NO_MAPPING;
122 }
123
124 if (PDPE->Bits.MustBeZero & 0x1) {
125 //
126 // 1GB PDPE
127 //
128 PTE1G = (PAGE_TABLE_1G_ENTRY *)PDPE;
129 Start = PTE1G->Uint64 & PAGING_1G_ADDRESS_MASK_64;
130 *PhysicalAddr = Start + VA.Pg1G.PhysPgOffset;
131 return EFI_SUCCESS;
132 }
133
134 //
135 // PDE
136 //
137 PDE = (PAGE_MAP_AND_DIRECTORY_POINTER *)(UINTN)(PDPE->Uint64 & PAGING_4K_ADDRESS_MASK_64);
138 PDE += VA.Pg4K.PDOffset;
139 VAStart.Pg4K.PDOffset = VA.Pg4K.PDOffset;
140 VAEnd.Pg4K.PDOffset = VA.Pg4K.PDOffset;
141
142 if (!PDE->Bits.Present) {
143 return EFI_NO_MAPPING;
144 }
145
146 if (PDE->Bits.MustBeZero & 0x1) {
147 //
148 // 2MB PDE
149 //
150 PTE2M = (PAGE_TABLE_2M_ENTRY *)PDE;
151 Start = PTE2M->Uint64 & PAGING_2M_ADDRESS_MASK_64;
152 *PhysicalAddr = Start + VA.Pg2M.PhysPgOffset;
153 return EFI_SUCCESS;
154 }
155
156 //
157 // PTE
158 //
159 PTE4K = (PAGE_TABLE_4K_ENTRY *)(UINTN)(PDE->Uint64 & PAGING_4K_ADDRESS_MASK_64);
160 PTE4K += VA.Pg4K.PTOffset;
161 VAStart.Pg4K.PTOffset = VA.Pg4K.PTOffset;
162 VAEnd.Pg4K.PTOffset = VA.Pg4K.PTOffset;
163
164 if (!PTE4K->Bits.Present) {
165 return EFI_NO_MAPPING;
166 }
167
168 Start = PTE4K->Uint64 & PAGING_4K_ADDRESS_MASK_64;
169 *PhysicalAddr = Start + VA.Pg4K.PhysPgOffset;
170
171 return EFI_SUCCESS;
172}

222VmMapVirtualPage (
223 IN OUT OC_VMEM_CONTEXT *Context,
224 IN OUT PAGE_MAP_AND_DIRECTORY_POINTER *PageTable OPTIONAL,
225 IN EFI_VIRTUAL_ADDRESS VirtualAddr,
226 IN EFI_PHYSICAL_ADDRESS PhysicalAddr
227 )
228{
229 EFI_PHYSICAL_ADDRESS Start;
230 VIRTUAL_ADDR VA;
231 VIRTUAL_ADDR VAStart;
232 VIRTUAL_ADDR VAEnd;
233 PAGE_MAP_AND_DIRECTORY_POINTER *PML4;
234 PAGE_MAP_AND_DIRECTORY_POINTER *PDPE;
235 PAGE_MAP_AND_DIRECTORY_POINTER *PDE;
236 PAGE_TABLE_4K_ENTRY *PTE4K;
237 PAGE_TABLE_4K_ENTRY *PTE4KTmp;
238 PAGE_TABLE_2M_ENTRY *PTE2M;
239 PAGE_TABLE_1G_ENTRY *PTE1G;
240 UINTN Index;
241 BOOLEAN WriteProtected;
242
243 if (PageTable == NULL) {
244 PageTable = OcGetCurrentPageTable (NULL);
245 }
246
247 WriteProtected = DisablePageTableWriteProtection ();
248
249 VA.Uint64 = (UINT64)VirtualAddr;
250
251 //
252 // PML4
253 //
254 PML4 = PageTable;
255 PML4 += VA.Pg4K.PML4Offset;
256
257 //
258 // There is a problem if our PML4 points to the same table as first PML4 entry
259 // since we may mess the mapping of first virtual region (happens in VBox and probably DUET).
260 // Check for this on first call and if true, just clear our PML4 - we'll rebuild at a later step.
261 //
262 if ( (PML4 != PageTable) && PML4->Bits.Present
263 && (PageTable->Bits.PageTableBaseAddress == PML4->Bits.PageTableBaseAddress))
264 {
265 PML4->Uint64 = 0;
266 }
267
268 VAStart.Uint64 = 0;
269 VAStart.Pg4K.PML4Offset = VA.Pg4K.PML4Offset;
270 VA_FIX_SIGN_EXTEND (VAStart);
271 VAEnd.Uint64 = ~(UINT64)0;
272 VAEnd.Pg4K.PML4Offset = VA.Pg4K.PML4Offset;
273 VA_FIX_SIGN_EXTEND (VAEnd);
274
275 if (!PML4->Bits.Present) {
276 PDPE = (PAGE_MAP_AND_DIRECTORY_POINTER *)VmAllocatePages (Context, 1);
277
278 if (PDPE == NULL) {
279 if (WriteProtected) {
280 EnablePageTableWriteProtection ();
281 }
282
283 return EFI_NO_MAPPING;
284 }
285
286 ZeroMem (PDPE, EFI_PAGE_SIZE);
287
288 //
289 // Init this whole 512 GB region with 512 1GB entry pages to map
290 // the first 512 GB physical space.
291 //
292 PTE1G = (PAGE_TABLE_1G_ENTRY *)PDPE;
293 Start = 0;
294 for (Index = 0; Index < 512; ++Index) {
295 PTE1G->Uint64 = Start & PAGING_1G_ADDRESS_MASK_64;
296 PTE1G->Bits.ReadWrite = 1;
297 PTE1G->Bits.Present = 1;
298 PTE1G->Bits.MustBe1 = 1;
299 PTE1G++;
300 Start += BASE_1GB;
301 }
302
303 //
304 // Put it to PML4.
305 //
306 PML4->Uint64 = ((UINT64)(UINTN)PDPE) & PAGING_4K_ADDRESS_MASK_64;
307 PML4->Bits.ReadWrite = 1;
308 PML4->Bits.Present = 1;
309 }
310
311 //
312 // PDPE
313 //
314 PDPE = (PAGE_MAP_AND_DIRECTORY_POINTER *)(UINTN)(PML4->Uint64 & PAGING_4K_ADDRESS_MASK_64);
315 PDPE += VA.Pg4K.PDPOffset;
316 VAStart.Pg4K.PDPOffset = VA.Pg4K.PDPOffset;
317 VAEnd.Pg4K.PDPOffset = VA.Pg4K.PDPOffset;
318
319 if (!PDPE->Bits.Present || (PDPE->Bits.MustBeZero & 0x1)) {
320 PDE = (PAGE_MAP_AND_DIRECTORY_POINTER *)VmAllocatePages (Context, 1);
321
322 if (PDE == NULL) {
323 if (WriteProtected) {
324 EnablePageTableWriteProtection ();
325 }
326
327 return EFI_NO_MAPPING;
328 }
329
330 ZeroMem (PDE, EFI_PAGE_SIZE);
331
332 if (PDPE->Bits.MustBeZero & 0x1) {
333 //
334 // This is 1 GB page. Init new PDE array to get the same
335 // mapping but with 2MB pages.
336 //
337 PTE2M = (PAGE_TABLE_2M_ENTRY *)PDE;
338 Start = PDPE->Uint64 & PAGING_1G_ADDRESS_MASK_64;
339
340 for (Index = 0; Index < 512; ++Index) {
341 PTE2M->Uint64 = Start & PAGING_2M_ADDRESS_MASK_64;
342 PTE2M->Bits.ReadWrite = 1;
343 PTE2M->Bits.Present = 1;
344 PTE2M->Bits.MustBe1 = 1;
345 PTE2M++;
346 Start += BASE_2MB;
347 }
348 }
349
350 //
351 // Put it to PDPE.
352 //
353 PDPE->Uint64 = ((UINT64)(UINTN)PDE) & PAGING_4K_ADDRESS_MASK_64;
354 PDPE->Bits.ReadWrite = 1;
355 PDPE->Bits.Present = 1;
356 }
357
358 //
359 // PDE
360 //
361 PDE = (PAGE_MAP_AND_DIRECTORY_POINTER *)(UINTN)(PDPE->Uint64 & PAGING_4K_ADDRESS_MASK_64);
362 PDE += VA.Pg4K.PDOffset;
363 VAStart.Pg4K.PDOffset = VA.Pg4K.PDOffset;
364 VAEnd.Pg4K.PDOffset = VA.Pg4K.PDOffset;
365
366 if (!PDE->Bits.Present || (PDE->Bits.MustBeZero & 0x1)) {
367 PTE4K = (PAGE_TABLE_4K_ENTRY *)VmAllocatePages (Context, 1);
368
369 if (PTE4K == NULL) {
370 if (WriteProtected) {
371 EnablePageTableWriteProtection ();
372 }
373
374 return EFI_NO_MAPPING;
375 }
376
377 ZeroMem (PTE4K, EFI_PAGE_SIZE);
378
379 if (PDE->Bits.MustBeZero & 0x1) {
380 //
381 // This is 2 MB page. Init new PTE array to get the same
382 // mapping but with 4KB pages.
383 //
384 PTE4KTmp = (PAGE_TABLE_4K_ENTRY *)PTE4K;
385 Start = PDE->Uint64 & PAGING_2M_ADDRESS_MASK_64;
386
387 for (Index = 0; Index < 512; ++Index) {
388 PTE4KTmp->Uint64 = Start & PAGING_4K_ADDRESS_MASK_64;
389 PTE4KTmp->Bits.ReadWrite = 1;
390 PTE4KTmp->Bits.Present = 1;
391 PTE4KTmp++;
392 Start += BASE_4KB;
393 }
394 }
395
396 //
397 // Put it to PDE.
398 //
399 PDE->Uint64 = ((UINT64)(UINTN)PTE4K) & PAGING_4K_ADDRESS_MASK_64;
400 PDE->Bits.ReadWrite = 1;
401 PDE->Bits.Present = 1;
402 }
403
404 //
405 // PTE
406 //
407 PTE4K = (PAGE_TABLE_4K_ENTRY *)(UINTN)(PDE->Uint64 & PAGING_4K_ADDRESS_MASK_64);
408 PTE4K += VA.Pg4K.PTOffset;
409 VAStart.Pg4K.PTOffset = VA.Pg4K.PTOffset;
410 VAEnd.Pg4K.PTOffset = VA.Pg4K.PTOffset;
411
412 //
413 // Put it to PTE.
414 //
415 PTE4K->Uint64 = ((UINT64)PhysicalAddr) & PAGING_4K_ADDRESS_MASK_64;
416 PTE4K->Bits.ReadWrite = 1;
417 PTE4K->Bits.Present = 1;
418
419 if (WriteProtected) {
420 EnablePageTableWriteProtection ();
421 }
422
423 return EFI_SUCCESS;
424}

426EFI_STATUS
427VmMapVirtualPages (
428 IN OUT OC_VMEM_CONTEXT *Context,
429 IN OUT PAGE_MAP_AND_DIRECTORY_POINTER *PageTable OPTIONAL,
430 IN EFI_VIRTUAL_ADDRESS VirtualAddr,
431 IN UINT64 NumPages,
432 IN EFI_PHYSICAL_ADDRESS PhysicalAddr
433 )
434{
435 EFI_STATUS Status;
436
437 if (PageTable == NULL) {
438 PageTable = OcGetCurrentPageTable (NULL);
439 }
440
441 Status = EFI_SUCCESS;
442
443 while (NumPages > 0 && !EFI_ERROR (Status)) {
444 Status = VmMapVirtualPage (
445 Context,
446 PageTable,
447 VirtualAddr,
448 PhysicalAddr
449 );
450
451 VirtualAddr += EFI_PAGE_SIZE;
452 PhysicalAddr += EFI_PAGE_SIZE;
453 NumPages--;
454 }
455
456 return Status;
457}
