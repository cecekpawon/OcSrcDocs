264InternalGetRecoveryOsBooter (

267 IN BOOLEAN Basic
268 )
269{

274 UINTN FilePathSize;

276 UINTN TmpPathSize;

278
279 Status = gBS->HandleProtocol (
280 Device,
281 &gEfiSimpleFileSystemProtocolGuid,
282 (VOID **)&FileSystem
283 );
284 if (EFI_ERROR (Status)) {
285 return Status;
286 }
287
288 Status = FileSystem->OpenVolume (FileSystem, &Root);
289 if (EFI_ERROR (Status)) {
290 return Status;
291 }
292
293 FilePathSize = 0;
294
295 if (!Basic) {
296 *FilePath = (EFI_DEVICE_PATH_PROTOCOL *)OcGetFileInfo (
297 Root,
298 &gAppleBlessedOsxFolderInfoGuid,
299 sizeof (EFI_DEVICE_PATH_PROTOCOL),
300 &FilePathSize
301 );
302 } else {
303 //
304 // Requested basic recovery support, i.e. only com.apple.recovery.boot folder check.
305 // This is useful for locating empty USB sticks with just a dmg in them.
306 //
307 *FilePath = NULL;
308 }
309
310 if (*FilePath != NULL) {
311 if (IsDevicePathValid (*FilePath, FilePathSize)) {
312 //
313 // We skip alternate entry when current one is the same.
314 // This is to prevent recovery and volume duplicates on HFS+ systems.
315 //
316
317 TmpPath = (EFI_DEVICE_PATH_PROTOCOL *)OcGetFileInfo (
318 Root,
319 &gAppleBlessedSystemFolderInfoGuid,
320 sizeof (EFI_DEVICE_PATH_PROTOCOL),
321 &TmpPathSize
322 );
323
324 if (TmpPath != NULL) {
325 if ( IsDevicePathValid (TmpPath, TmpPathSize)
326 && IsDevicePathEqual (TmpPath, *FilePath))
327 {
328 DEBUG ((DEBUG_INFO, "Skipping equal alternate device path %p\n", Device));
329 Status = EFI_ALREADY_STARTED;
330 FreePool (*FilePath);
331 *FilePath = NULL;
332 }
333
334 FreePool (TmpPath);
335 }
336
337 if (!EFI_ERROR (Status)) {
338 //
339 // This entry should point to a folder with recovery.
340 // Apple never adds trailing slashes to blessed folder paths.
341 // However, we do rely on trailing slashes in folder paths and add them here.
342 //
343 TmpPath = TrailedBooterDevicePath (*FilePath);
344 if (TmpPath != NULL) {
345 FreePool (*FilePath);
346 *FilePath = TmpPath;
347 }
348 }
349 } else {
350 FreePool (*FilePath);
351 *FilePath = NULL;
352 Status = EFI_NOT_FOUND;
353 }
354 } else {
355 //
356 // Ok, this one can still be FileVault 2 HFS+ recovery or just a hardcoded basic recovery.
357 // Apple does add its path to so called "Alternate OS blessed file/folder", but this
358 // path is not accessible from HFSPlus.efi driver. Just why???
359 // Their SlingShot.efi app just bruteforces com.apple.recovery.boot directory existence,
360 // and we have to copy.
361 //
362 Status = OcSafeFileOpen (Root, &Recovery, L"\\com.apple.recovery.boot", EFI_FILE_MODE_READ, 0);
363 if (!EFI_ERROR (Status)) {
364 //
365 // Do not do any extra checks for simplicity, as they will be done later either way.
366 //
367 Root->Close (Recovery);
368 Status = EFI_NOT_FOUND;
369 TmpPath = DevicePathFromHandle (Device);
370
371 if (TmpPath != NULL) {
372 *FilePath = AppendFileNameDevicePath (TmpPath, L"\\com.apple.recovery.boot\\");
373 if (*FilePath != NULL) {
374 DEBUG_CODE_BEGIN ();
375 DevicePathText = ConvertDevicePathToText (*FilePath, FALSE, FALSE);
376 if (DevicePathText != NULL) {
377 DEBUG ((DEBUG_INFO, "OCB: Got recovery dp %s\n", DevicePathText));
378 FreePool (DevicePathText);
379 }
380
381 DEBUG_CODE_END ();
382 Status = EFI_SUCCESS;
383 }
384 }
385 } else {
386 Status = EFI_NOT_FOUND;
387 }
388 }
389
390 Root->Close (Root);
391
392 return Status;
393}

550// Duplicate each flavour in list, w/ apple version added to first of each duplicate
551//
552STATIC
553CHAR8 *
554InternalAddAppleVersion (
555 IN CHAR8 *Flavour,
556 IN CHAR8 *Version
557 )
558{
559 UINTN VersionLength;
560 UINTN FlavourLength;
561 UINTN SepCount;
562 UINTN Size;
563 UINTN Index;
564 CHAR8 *NewFlavour;
565 CHAR8 *Start;
566 CHAR8 *End;
567 CHAR8 *Pos;
568
569 ASSERT (Flavour != NULL);
570 ASSERT (Version != NULL);
571
572 VersionLength = AsciiStrLen (Version);
573 FlavourLength = AsciiStrLen (Flavour);
574 SepCount = 0;
575 for (Index = 0; Index < FlavourLength; Index++) {
576 if (Flavour[Index] == ':') {
577 ++SepCount;
578 }
579 }
580
581 Size = 2 * (FlavourLength + 1) + VersionLength * (SepCount + 1);
582
583 if (Size > OC_MAX_CONTENT_FLAVOUR_SIZE) {
584 return Flavour;
585 }
586
587 NewFlavour = AllocatePool (Size);
588 if (NewFlavour == NULL) {
589 return Flavour;
590 }
591
592 Pos = NewFlavour;
593 End = Flavour - 1;
594 do {
595 for (Start = ++End; *End != '\0' && *End != ':'; ++End) {
596 }
597
598 AsciiStrnCpyS (Pos, Size - (Pos - NewFlavour), Start, End - Start);
599 Pos += End - Start;
600 AsciiStrnCpyS (Pos, Size - (Pos - NewFlavour), Version, VersionLength);
601 Pos += VersionLength;
602 *Pos++ = ':';
603 AsciiStrnCpyS (Pos, Size - (Pos - NewFlavour), Start, End - Start);
604 Pos += End - Start;
605 *Pos++ = ':';
606 } while (*End != '\0');
607
608 *(Pos - 1) = '\0';
609
610 ASSERT ((UINTN)(Pos - NewFlavour) == Size);
611
612 FreePool (Flavour);
613
614 return NewFlavour;
615}

617EFI_STATUS
618InternalDescribeBootEntry (
619 IN OC_BOOT_CONTEXT *BootContext,
620 IN OUT OC_BOOT_ENTRY *BootEntry
621 )
622{
623 EFI_STATUS Status;
624 CHAR16 *BootDirectoryName;
625 CHAR16 *TmpBootName;
626 EFI_HANDLE Device;
627 UINT32 BcdSize;
628 EFI_SIMPLE_FILE_SYSTEM_PROTOCOL *FileSystem;
629 CHAR8 *ContentFlavour;
630 CHAR8 AppleVersion[OC_APPLE_VERSION_MAX_SIZE];
631 CHAR8 *Dot;
632
633 AppleVersion[0] = '\0';
634
635 //
636 // Custom entries need no special description.
637 //
638 if ((BootEntry->Type == OC_BOOT_EXTERNAL_OS) || (BootEntry->Type == OC_BOOT_EXTERNAL_TOOL)) {
639 return EFI_SUCCESS;
640 }
641
642 Status = OcBootPolicyDevicePathToDirPath (
643 BootEntry->DevicePath,
644 &BootDirectoryName,
645 &Device
646 );
647
648 if (EFI_ERROR (Status)) {
649 return Status;
650 }
651
652 Status = gBS->HandleProtocol (
653 Device,
654 &gEfiSimpleFileSystemProtocolGuid,
655 (VOID **)&FileSystem
656 );
657
658 if (EFI_ERROR (Status)) {
659 FreePool (BootDirectoryName);
660 return Status;
661 }
662
663 //
664 // Try to use APFS-style label or legacy HFS one.
665 //
666 BootEntry->Name = InternalGetAppleDiskLabel (FileSystem, BootDirectoryName, L".contentDetails");
667 if (BootEntry->Name == NULL) {
668 BootEntry->Name = InternalGetAppleDiskLabel (FileSystem, BootDirectoryName, L".disk_label.contentDetails");
669 }
670
671 //
672 // With FV2 encryption on HFS+ the actual boot happens from "Recovery HD/S/L/CoreServices".
673 // For some reason "Recovery HD/S/L/CoreServices/.disk_label" may not get updated immediately,
674 // and will contain "Recovery HD" despite actually pointing to "Macintosh HD".
675 // This also spontaneously happens with renamed APFS volumes. The workaround is to manually
676 // edit the file or sometimes choose the boot volume once more in preferences.
677 //
678 // TODO: Bugreport this to Apple, as this is clearly their bug, which should be reproducible
679 // on original hardware.
680 //
681 // There exists .root_uuid, which contains real partition UUID in ASCII, however, Apple
682 // BootPicker only uses it for entry deduplication, and we cannot figure out the name
683 // on an encrypted volume anyway.
684 //
685
686 //
687 // Windows boot entry may have a custom name, so ensure OC_BOOT_WINDOWS is set correctly.
688 //
689 if ((BootEntry->Type == OC_BOOT_UNKNOWN) && BootEntry->IsGeneric) {
690 DEBUG ((DEBUG_INFO, "OCB: Trying to detect Microsoft BCD\n"));
691 Status = OcReadFileSize (FileSystem, L"\\EFI\\Microsoft\\Boot\\BCD", &BcdSize);
692 if (!EFI_ERROR (Status)) {
693 BootEntry->Type = OC_BOOT_WINDOWS;
694 }
695 }
696
697 if ((BootEntry->Type == OC_BOOT_WINDOWS) && (BootEntry->Name == NULL)) {
698 BootEntry->Name = AllocateCopyPool (sizeof (L"Windows"), L"Windows");
699 }
700
701 if (BootEntry->Name == NULL) {
702 //
703 // Special case - installer should be clearly identified to end users but does not normally
704 // contain text label, only pre-rendered graphical label which is not usable in builtin
705 // picker, or in Canopy with disk labels disabled.
706 //
707 if (StrStr (BootDirectoryName, L"com.apple.installer") != NULL) {
708 BootEntry->Name = AllocateCopyPool (L_STR_SIZE (L"macOS Installer"), L"macOS Installer");
709 } else {
710 BootEntry->Name = OcGetVolumeLabel (FileSystem);
711 if (BootEntry->Name != NULL) {
712 if ( (StrCmp (BootEntry->Name, L"Recovery HD") == 0)
713 || (StrCmp (BootEntry->Name, L"Recovery") == 0))
714 {
715 if ((BootEntry->Type == OC_BOOT_UNKNOWN) || (BootEntry->Type == OC_BOOT_APPLE_OS)) {
716 BootEntry->Type = OC_BOOT_APPLE_RECOVERY;
717 }
718
719 Status = InternalGetAppleVersion (FileSystem, BootDirectoryName, AppleVersion);
720 if (EFI_ERROR (Status)) {
721 TmpBootName = NULL;
722 } else {
723 TmpBootName = InternalGetAppleRecoveryName (AppleVersion);
724 }
725 } else if (StrCmp (BootEntry->Name, L"Preboot") == 0) {
726 //
727 // Common Big Sur beta bug failing to create .contentDetails files.
728 // Workaround it by using the standard installed macOS system volume name.
729 // Applies to anything on the system volume without text labels (and not already
730 // handled above, such as installer).
731 //
732 TmpBootName = AllocateCopyPool (sizeof (L"Macintosh HD"), L"Macintosh HD");
733 } else {
734 TmpBootName = NULL;
735 }
736
737 if (TmpBootName != NULL) {
738 FreePool (BootEntry->Name);
739 BootEntry->Name = TmpBootName;
740 }
741 }
742 }
743 }
744
745 if (BootEntry->Name == NULL) {
746 FreePool (BootDirectoryName);
747 return EFI_NOT_FOUND;
748 }
749
750 BootEntry->PathName = BootDirectoryName;
751
752 //
753 // Get user-specified or builtin content flavour.
754 //
755 if ((BootContext->PickerContext->PickerAttributes & OC_ATTR_USE_FLAVOUR_ICON) != 0) {
756 BootEntry->Flavour = InternalGetContentFlavour (FileSystem, BootDirectoryName, L".contentFlavour");
757 }
758
759 if ((BootEntry->Flavour == NULL) || (AsciiStrCmp (BootEntry->Flavour, OC_FLAVOUR_AUTO) == 0)) {
760 switch (BootEntry->Type) {
761 case OC_BOOT_APPLE_OS:
762 ContentFlavour = AllocateCopyPool (sizeof (OC_FLAVOUR_APPLE_OS), OC_FLAVOUR_APPLE_OS);
763 if ((BootContext->PickerContext->PickerAttributes & OC_ATTR_USE_FLAVOUR_ICON) != 0) {
764 InternalGetAppleVersion (FileSystem, BootDirectoryName, AppleVersion);
765 }
766
767 break;
768 case OC_BOOT_APPLE_FW_UPDATE:
769 ContentFlavour = AllocateCopyPool (sizeof (OC_FLAVOUR_APPLE_FW), OC_FLAVOUR_APPLE_FW);
770 break;
771 case OC_BOOT_APPLE_RECOVERY:
772 ContentFlavour = AllocateCopyPool (sizeof (OC_FLAVOUR_APPLE_RECOVERY), OC_FLAVOUR_APPLE_RECOVERY);
773 break;
774 case OC_BOOT_APPLE_TIME_MACHINE:
775 ContentFlavour = AllocateCopyPool (sizeof (OC_FLAVOUR_APPLE_TIME_MACHINE), OC_FLAVOUR_APPLE_TIME_MACHINE);
776 break;
777 case OC_BOOT_WINDOWS:
778 ContentFlavour = AllocateCopyPool (sizeof (OC_FLAVOUR_WINDOWS), OC_FLAVOUR_WINDOWS);
779 break;
780 case OC_BOOT_UNKNOWN:
781 ContentFlavour = NULL;
782 break;
783 default:
784 DEBUG ((DEBUG_ERROR, "OCB: Entry kind %d unsupported for flavour\n", BootEntry->Type));
785 ContentFlavour = NULL;
786 break;
787 }
788
789 if ((BootEntry->Type & OC_BOOT_APPLE_ANY) != 0) {
790 if (ContentFlavour == NULL) {
791 ASSERT (FALSE);
792 } else if ( (AppleVersion[0] != '\0')
793 && ((BootContext->PickerContext->PickerAttributes & OC_ATTR_USE_FLAVOUR_ICON) != 0)
794 )

796 //
