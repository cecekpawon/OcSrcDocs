41RETURN_STATUS
42PeCoffGetSecurityDirectoryEntry (
43 IN PE_COFF_IMAGE_CONTEXT *Context,
44 IN UINT32 FileSize,
45 OUT CONST EFI_IMAGE_DATA_DIRECTORY **DirectoryEntry
46 )
47{
48 CONST EFI_IMAGE_NT_HEADERS32 *Pe32Hdr;
49 CONST EFI_IMAGE_NT_HEADERS64 *Pe32PlusHdr;
50 UINT32 EntryTop;
51 BOOLEAN Result;
52
53 ASSERT (Context != NULL);
54 ASSERT (DirectoryEntry != NULL);
55
56 switch (Context->ImageType) {
57 case ImageTypePe32:
58 Pe32Hdr = (CONST EFI_IMAGE_NT_HEADERS32 *)(CONST VOID *)(
59 (CONST CHAR8 *)Context->FileBuffer + Context->ExeHdrOffset
60 );
61
62 if (Pe32Hdr->NumberOfRvaAndSizes <= EFI_IMAGE_DIRECTORY_ENTRY_SECURITY) {
63 return RETURN_UNSUPPORTED;
64 }
65
66 *DirectoryEntry = &Pe32Hdr->DataDirectory[EFI_IMAGE_DIRECTORY_ENTRY_SECURITY];
67 break;
68
69 case ImageTypePe32Plus:
70 Pe32PlusHdr = (CONST EFI_IMAGE_NT_HEADERS64 *)(CONST VOID *)(
71 (CONST CHAR8 *)Context->FileBuffer + Context->ExeHdrOffset
72 );
73
74 if (Pe32PlusHdr->NumberOfRvaAndSizes <= EFI_IMAGE_DIRECTORY_ENTRY_SECURITY) {
75 return RETURN_UNSUPPORTED;
76 }
77
78 *DirectoryEntry = &Pe32PlusHdr->DataDirectory[EFI_IMAGE_DIRECTORY_ENTRY_SECURITY];
79 break;
80
81 default:
82 //
83 // TE entries do not have SecDir.
84 //
85 return RETURN_INVALID_PARAMETER;
86 }
87
88 //
89 // Security entry must be outside of the headers.
90 //
91 if ((*DirectoryEntry)->VirtualAddress < Context->SizeOfHeaders) {
92 return RETURN_INVALID_PARAMETER;
93 }
94
95 Result = OcOverflowAddU32 (
96 (*DirectoryEntry)->VirtualAddress,
97 (*DirectoryEntry)->Size,
98 &EntryTop
99 );
100 if (Result || (EntryTop > FileSize)) {
101 return RETURN_INVALID_PARAMETER;
102 }
103
104 return RETURN_SUCCESS;
105}

158 //
159 *SignedFileSize = SecDir->VirtualAddress;

164STATIC
165EFI_STATUS
166PeCoffGetAppleSignature (
167 IN PE_COFF_IMAGE_CONTEXT *Context,
168 IN APPLE_EFI_CERTIFICATE_INFO *CertInfo,
169 OUT APPLE_SIGNATURE_CONTEXT *SignatureContext
170 )
171{
172 UINTN Index;
173 UINT8 PublicKeyHash[SHA256_DIGEST_SIZE];
174 APPLE_EFI_CERTIFICATE *Cert;
175
176 //
177 // Check that certificate is expected.
178 //
179 if (CertInfo->CertSize != sizeof (APPLE_EFI_CERTIFICATE)) {
180 DEBUG ((DEBUG_INFO, "OCPE: Certificate has invalid size %u\n", CertInfo->CertSize));
181 return EFI_UNSUPPORTED;
182 }
183
184 if (!OC_TYPE_ALIGNED (APPLE_EFI_CERTIFICATE, CertInfo->CertOffset)) {
185 DEBUG ((DEBUG_INFO, "OCPE: Certificate is misaligned %X\n", CertInfo->CertOffset));
186 return EFI_UNSUPPORTED;
187 }
188
189 //
190 // Extract signature directory
191 //
192 Cert = (APPLE_EFI_CERTIFICATE *)(
193 (UINT8 *)Context->FileBuffer + CertInfo->CertOffset
194 );
195
196 //
197 // Compare size of signature directory with value from PE SecDir header
198 //
199 if (CertInfo->CertSize != Cert->CertSize) {
200 DEBUG ((DEBUG_INFO, "OCPE: Certificate size mismatch %u vs %u\n", CertInfo->CertSize, Cert->CertSize));
201 return EFI_UNSUPPORTED;
202 }
203
204 //
205 // Verify certificate type
206 //
207 if (Cert->CertType != APPLE_EFI_CERTIFICATE_TYPE) {
208 DEBUG ((DEBUG_INFO, "OCPE: Unknown certificate type %u\n", Cert->CertType));
209 return EFI_UNSUPPORTED;
210 }
211
212 //
213 // Verify certificate GUID
214 //
215 if (!CompareGuid (&Cert->AppleSignatureGuid, &gAppleEfiCertificateGuid)) {
216 DEBUG ((DEBUG_INFO, "OCPE: Unknown certificate signature %g\n", Cert->AppleSignatureGuid));
217 return EFI_UNSUPPORTED;
218 }
219
220 //
221 // Verify HashType == Rsa2048Sha256
222 //
223 if (!CompareGuid (&Cert->CertData.HashType, &gEfiCertTypeRsa2048Sha256Guid)) {
224 DEBUG ((DEBUG_INFO, "OCPE: Unknown certificate hash %g\n", Cert->CertData.HashType));
225 return EFI_UNSUPPORTED;
226 }
227
228 //
229 // Calculate public key hash and find it.
230 //
231 Sha256 (
232 PublicKeyHash,
233 Cert->CertData.PublicKey,
234 sizeof (Cert->CertData.PublicKey)
235 );
236
237 //
238 // Verify public key existence in the database and store it in the context.
239 //
240 for (Index = 0; Index < NUM_OF_PK; ++Index) {
241 if (CompareMem (PkDataBase[Index].Hash, PublicKeyHash, sizeof (PublicKeyHash)) == 0) {
242 SignatureContext->PublicKey = (OC_RSA_PUBLIC_KEY *)PkDataBase[Index].PublicKey;
243 break;
244 }
245 }
246
247 if (Index == NUM_OF_PK) {
248 DEBUG ((DEBUG_INFO, "OCPE: Unknown publickey or malformed certificate\n"));
249 return EFI_UNSUPPORTED;
250 }
251
252 //
253 // Convert signature to big endian and store it in the context as well.
254 //
255 for (Index = 0; Index < sizeof (Cert->CertData.PublicKey); Index++) {
256 SignatureContext->Signature[sizeof (Cert->CertData.PublicKey) - 1 - Index]
257 = Cert->CertData.Signature[Index];
258 }
259
260 return EFI_SUCCESS;
261}

263STATIC
264EFI_STATUS
265PeCoffSanitiseAppleImage (
266 IN PE_COFF_IMAGE_CONTEXT *Context,
267 IN UINT32 SecDirOffset,
268 IN UINT32 SignedFileSize,
269 IN UINT32 FileSize
270 )
271{
272 //
273 // While TE files cannot technically have SecDir,
274 // one might add more PE types in the future technically.
275 // Restrict file type as early as possible.
276 //
277 if ( (Context->ImageType != ImageTypePe32)
278 && (Context->ImageType != ImageTypePe32Plus))
279 {
280 DEBUG ((DEBUG_INFO, "OCPE: Unsupported image type %d for Apple Image\n", Context->ImageType));
281 return EFI_UNSUPPORTED;
282 }
283
284 //
285 // Apple images are required to start with DOS header.
286 // DOS header is really the only reason ExeHdrOffset can be
287 // non-zero, do not bother extra checking.
288 //
289 if (Context->ExeHdrOffset < sizeof (EFI_IMAGE_DOS_HEADER)) {
290 DEBUG ((DEBUG_INFO, "OCPE: DOS header is required for signed Apple Image\n"));
291 return EFI_UNSUPPORTED;
292 }
293
294 //
295 // Zero memory between DOS and optional header.
296 //
297 ZeroMem (
298 (UINT8 *)Context->FileBuffer + sizeof (EFI_IMAGE_DOS_HEADER),
299 Context->ExeHdrOffset - sizeof (EFI_IMAGE_DOS_HEADER)
300 );
301
302 //
303 // Zero checksum as we do not hash it.
304 //
305 ZeroMem (
306 (UINT8 *)Context->FileBuffer + Context->ExeHdrOffset + APPLE_CHECKSUM_OFFSET,
307 APPLE_CHECKSUM_SIZE
308 );
309
310 //
311 // Zero sec entry as we do not hash it as well.
312 //
313 ZeroMem (
314 (UINT8 *)Context->FileBuffer + SecDirOffset,
315 sizeof (EFI_IMAGE_DATA_DIRECTORY)
316 );
317
318 //
319 // Zero signature as we do not hash it.
320 //
321 ZeroMem (
322 (UINT8 *)Context->FileBuffer + SignedFileSize,
323 FileSize - SignedFileSize
324 );
325
326 return EFI_SUCCESS;
327}

329STATIC
330VOID
331PeCoffHashAppleImage (
332 IN PE_COFF_IMAGE_CONTEXT *Context,
333 IN UINT32 SecDirOffset,
334 IN UINT32 SignedFileSize,
335 OUT UINT8 *Hash
336 )
337{
338 UINTN HashSize;
339 UINT8 *HashBase;
340 SHA256_CONTEXT HashContext;
341
342 //
343 // Initialise a SHA hash context
344 //
345 Sha256Init (&HashContext);
346
347 //
348 // Hash DOS header and without DOS stub.
349 //
350 HashBase = (UINT8 *)Context->FileBuffer;
351 HashSize = sizeof (EFI_IMAGE_DOS_HEADER);
352 Sha256Update (&HashContext, HashBase, HashSize);
353
354 //
355 // Hash optional header prior to checksum.
356 //
357 HashBase += Context->ExeHdrOffset;
358 HashSize = APPLE_CHECKSUM_OFFSET;
359 Sha256Update (&HashContext, HashBase, HashSize);
360
361 //
362 // Hash the rest of the header up to security directory.
363 //
364 HashBase += HashSize + APPLE_CHECKSUM_SIZE;
365 HashSize = SecDirOffset - (UINT32)(HashBase - (UINT8 *)Context->FileBuffer);
366 Sha256Update (&HashContext, HashBase, HashSize);
367
368 //
369 // Hash the rest of the image skipping security directory.
370 //
371 HashBase += HashSize + sizeof (EFI_IMAGE_DATA_DIRECTORY);
372 HashSize = SignedFileSize - (UINT32)(HashBase - (UINT8 *)Context->FileBuffer);
373 Sha256Update (&HashContext, HashBase, HashSize);
374
375 Sha256Final (&HashContext, Hash);
376}

464EFI_STATUS
465PeCoffGetApfsDriverVersion (
466 IN VOID *DriverBuffer,
467 IN UINT32 DriverSize,
468 OUT APFS_DRIVER_VERSION **DriverVersionPtr
469 )
470{
471 //
472 // apfs.efi versioning is more restricted than generic PE parsing.
473 //
474
475 EFI_STATUS ImageStatus;
476 PE_COFF_IMAGE_CONTEXT ImageContext;
477 EFI_IMAGE_NT_HEADERS64 *OptionalHeader;
478 EFI_IMAGE_SECTION_HEADER *SectionHeader;
479 APFS_DRIVER_VERSION *DriverVersion;
480 UINT32 ImageVersion;
481
482 ImageStatus = PeCoffInitializeContext (
483 &ImageContext,
484 DriverBuffer,
485 DriverSize
486 );
487 if (EFI_ERROR (ImageStatus)) {
488 DEBUG ((DEBUG_INFO, "OCPE: PeCoff init apfs failure - %r\n", ImageStatus));
489 return EFI_UNSUPPORTED;
490 }
491
492 if ( (ImageContext.Machine != IMAGE_FILE_MACHINE_X64)
493 || (ImageContext.ImageType != ImageTypePe32Plus)
494 || (ImageContext.Subsystem != EFI_IMAGE_SUBSYSTEM_EFI_BOOT_SERVICE_DRIVER))
495 {
496 DEBUG ((DEBUG_INFO, "OCPE: PeCoff unsupported image\n"));
497 return EFI_UNSUPPORTED;
498 }
499
500 //
501 // Get image version. The header is already verified for us.
502 //
503 OptionalHeader = (EFI_IMAGE_NT_HEADERS64 *)(
504 (CONST UINT8 *)ImageContext.FileBuffer
505 + ImageContext.ExeHdrOffset
506 );
507 ImageVersion = (UINT32)OptionalHeader->MajorImageVersion << 16
508 | (UINT32)OptionalHeader->MinorImageVersion;
509
510 //
511 // Get .text contents. The sections are already verified for us,
512 // but it can be smaller than APFS version.
513 //
514 SectionHeader = (EFI_IMAGE_SECTION_HEADER *)(
515 (CONST UINT8 *)ImageContext.FileBuffer
516 + ImageContext.SectionsOffset
517 );
518
519 if (AsciiStrnCmp ((CHAR8 *)SectionHeader->Name, ".text", sizeof (SectionHeader->Name)) != 0) {
520 return EFI_UNSUPPORTED;
521 }
522
523 if (sizeof (APFS_DRIVER_VERSION) > SectionHeader->SizeOfRawData) {
524 return EFI_BUFFER_TOO_SMALL;
525 }
526
527 //
528 // Finally get driver version.
529 //
530 DriverVersion = (APFS_DRIVER_VERSION *)(
531 (CONST UINT8 *)ImageContext.FileBuffer
532 + SectionHeader->PointerToRawData
533 );
534
535 if ( (DriverVersion->Magic != APFS_DRIVER_VERSION_MAGIC)
536 || (DriverVersion->ImageVersion != ImageVersion))
537 {
538 return EFI_INVALID_PARAMETER;
539 }
540
541 *DriverVersionPtr = DriverVersion;
542 return EFI_SUCCESS;
543}
