80ToggleSipGetBootEntries (

87 EFI_STATUS Status;
88 UINT32 CsrActiveConfig;
89 BOOLEAN IsEnabled;
90
91 //
92 // Custom entries only.
93 //
94 if (Device != NULL) {
95 return EFI_NOT_FOUND;
96 }
97
98 //
99 // Configure current action.
100 //
101 Status = OcGetSip (&CsrActiveConfig, &mAttributes);
102 if (!EFI_ERROR (Status) || (Status == EFI_NOT_FOUND)) {
103 IsEnabled = OcIsSipEnabled (Status, CsrActiveConfig);
104
105 //
106 // Same logic as in CsrUtil.efi to determine attributes to use
107 // (specifically keep same V or NV setting as existing value, if any).
108 //
109 if (Status == EFI_NOT_FOUND) {
110 //
111 // TODO: We may want to upgrade Boot Entry Protocol again, so that
112 // this line is able to access and respect OC WriteFlash setting?
113 //
114 mAttributes = CSR_APPLE_SIP_NVRAM_NV_ATTR;
115 CsrActiveConfig = 0;
116 } else {
117 //
118 // We are finding other bits set on Apl, specifically 0x80000000,
119 // so only consider relevant bits.
120 //
121 mAttributes &= CSR_APPLE_SIP_NVRAM_NV_ATTR;
122 }
123 } else {
124 DEBUG ((DEBUG_WARN, "BEP: ToggleSip failed to read csr-active-config, aborting! - %r\n", Status));
125 return Status;
126 }
127
128 if (IsEnabled) {
129 mToggleSipBootEntries[0].Name = OC_MENU_SIP_IS_ENABLED;
130 mToggleSipBootEntries[0].Flavour = OC_FLAVOUR_TOGGLE_SIP_ENABLED;
131 mToggleSipBootEntries[0].AudioBasePath = OC_VOICE_OVER_AUDIO_FILE_SIP_IS_ENABLED;
132 mCsrNextConfig = mCsrUserConfig;
133 } else {
134 mToggleSipBootEntries[0].Name = OC_MENU_SIP_IS_DISABLED;
135 mToggleSipBootEntries[0].Flavour = OC_FLAVOUR_TOGGLE_SIP_DISABLED;
136 mToggleSipBootEntries[0].AudioBasePath = OC_VOICE_OVER_AUDIO_FILE_SIP_IS_DISABLED;
137 mCsrNextConfig = 0;
138 }
139
140 DEBUG ((
141 DEBUG_INFO,
142 "BEP: Toggle SIP entry, currently %a, will change 0x%X->0x%X%a\n",
143 (IsEnabled ? "enabled" : "disabled"),
144 CsrActiveConfig,
145 mCsrNextConfig,
146 ((mAttributes & EFI_VARIABLE_NON_VOLATILE) == 0 ? " (volatile)" : "")
147 ));
148
149 *Entries = mToggleSipBootEntries;
150 *NumEntries = ARRAY_SIZE (mToggleSipBootEntries);
151
152 return EFI_SUCCESS;
153}

183 Status = EFI_NOT_FOUND;

185 if (OcHasLoadOptions (LoadedImage->LoadOptionsSize, LoadedImage->LoadOptions)) {

191

194 }
195 }
196
197 if (!EFI_ERROR (Status)) {
198 mCsrUserConfig = (UINT32)Data;
199 if (OcIsSipEnabled (EFI_SUCCESS, mCsrUserConfig)) {
200 DEBUG ((DEBUG_WARN, "BEP: Specified value 0x%X will not disable SIP!\n", mCsrUserConfig));
201 }
202 } else {
203 mCsrUserConfig = OC_CSR_DISABLE_FLAGS;
204 }
205
206 Status = gBS->InstallMultipleProtocolInterfaces (
207 &ImageHandle,
208 &gOcBootEntryProtocolGuid,
209 &mToggleSipBootEntryProtocol,

211 );
